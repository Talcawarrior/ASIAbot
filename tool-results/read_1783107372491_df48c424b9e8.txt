   156→async def broadcast_message(message: dict):
   157→    """Broadcast message to all connected WebSocket clients."""
   158→    if not state.websocket_clients:
   159→        return
   160→    disconnected = []
   161→    for client in state.websocket_clients:
   162→        try:
   163→            await client.send_json(message)
   164→        except Exception:
   165→            disconnected.append(client)
   166→    for client in disconnected:
   167→        if client in state.websocket_clients:
   168→            state.websocket_clients.remove(client)

   184→@app.get("/api/status")
   185→def get_status():
   186→    """Get bot status and metrics with strict accounting."""
   187→    from sqlalchemy import func
   188→
   189→    from database.models import Analysis, Bet
   190→
   191→    db = get_db_session()
   192→    try:
   193→        db.query(Portfolio).filter(Portfolio.id == 1).first()
   194→
   195→        # 1. Realized PnL (Closed bets)
   196→        from datetime import datetime, timezone
   197→
   198→        _ts = datetime.now(timezone.utc).replace(tzinfo=None)
   199→        _today_start = _ts.replace(hour=0, minute=0, second=0, microsecond=0)
   200→        # All closed bet statuses (settled + bot-closed early exits)
   201→        _closed_statuses = ("won", "lost", "settled", "closed_early")
   202→        from sqlalchemy import or_
   203→
   204→        daily_pnl = (
   205→            db.query(func.coalesce(func.sum(Bet.pnl), 0.0))
   206→            .filter(
   207→                Bet.status.in_(_closed_statuses),
   208→                or_(Bet.settled_at >= _today_start, Bet.closed_at >= _today_start),
   209→            )
   210→            .scalar()
   211→        ) or 0.0
   212→
   213→        realized_pnl_db = (
   214→            db.query(func.coalesce(func.sum(Bet.pnl), 0.0)).filter(Bet.status.in_(_closed_statuses)).scalar()
   215→        ) or 0.0
   216→
   217→        # 2. Unrealized PnL (Open bets)
   218→        open_statuses = OPEN_BET_STATUSES
   219→        unrealized_pnl_db = (
   220→            db.query(func.coalesce(func.sum(Bet.unrealized_pnl), 0.0)).filter(Bet.status.in_(open_statuses)).scalar()
   221→        ) or 0.0
   222→
   223→        # 3. Counts — win/loss based on PnL (includes closed_early)
   224→        _all_closed = db.query(Bet.pnl).filter(Bet.status.in_(_closed_statuses)).all()
   225→        win_count = sum(1 for b in _all_closed if (b.pnl or 0) > 0)
   226→        loss_count = sum(1 for b in _all_closed if (b.pnl or 0) <= 0)
   227→        total_bets_db = db.query(Bet).filter(Bet.status.in_(open_statuses)).count()
   228→        total_signals_db = db.query(Analysis).filter(Analysis.should_bet.is_(True)).count()
   229→
   230→        # Open positions with details for frontend
   231→        open_bets = db.query(Bet).filter(Bet.status.in_(open_statuses)).all()
   232→        open_positions = []
   233→        for bet in open_bets:
   234→            wm = db.query(WeatherMarket).filter(WeatherMarket.id == bet.market_id).first()
   235→            open_positions.append(
   236→                {
   237→                    "id": str(bet.id),
   238→                    "city": bet.city,
   239→                    "side": bet.side,
   240→                    "entry_price": float(bet.entry_price or 0),
   241→                    "current_price": float(bet.current_price or bet.entry_price or 0),
   242→                    "unrealized_pnl": float(bet.unrealized_pnl or 0),
   243→                    "edge": float(bet.expected_value or 0) * 100,
   244→                    "shares": float(bet.shares or 0),
   245→                    "amount": float(bet.amount or 0),
   246→                    "opened_at": bet.placed_at.isoformat() if bet.placed_at else None,
   247→                    "market_id": bet.market_id,
   248→                    "market_type": wm.market_type if wm else None,
   249→                    "threshold": float(wm.threshold) if wm and wm.threshold else None,
   250→                    "question": wm.question if wm else None,
   251→                    "metric": wm.metric if wm else None,
   252→                }
   253→            )
   254→
   255→        exposure_db = (
   256→            db.query(func.coalesce(func.sum(Bet.amount), 0.0)).filter(Bet.status.in_(open_statuses)).scalar()
   257→        ) or 0.0
   258→
   259→        initial_capital = state.config.INITIAL_PORTFOLIO
   260→        total_pnl = realized_pnl_db + unrealized_pnl_db
   261→
   262→        # Total amount staked in settled bets (sum of all bet amounts
   263→        # regardless of win/loss). ROI = PnL / total_stake, NOT PnL / initial.
   264→        total_stake_settled = (
   265→            db.query(func.coalesce(func.sum(Bet.amount), 0.0)).filter(Bet.status.in_(_closed_statuses)).scalar()
   266→        ) or 0.0
   267→
   268→        # Realized PnL from bets closed BEFORE today (for daily exposure cap)
   269→        realized_before_today = (
   270→            db.query(func.coalesce(func.sum(Bet.pnl), 0.0))
   271→            .filter(
   272→                Bet.status.in_(_closed_statuses),
   273→                or_(
   274→                    Bet.settled_at < _today_start,
   275→                    Bet.closed_at < _today_start,
   276→                ),
   277→            )
   278→            .scalar()
   279→        ) or 0.0
   280→
   281→        # ROI for CLOSED bets: realized PNL / total stake (bet amounts)
   282→        total_roi = (realized_pnl_db / total_stake_settled) * 100 if total_stake_settled > 0 else 0
   283→        # Daily ROI: daily realized PNL / total stake settled today
   284→        total_stake_today = (
   285→            db.query(func.coalesce(func.sum(Bet.amount), 0.0))
   286→            .filter(
   287→                Bet.status.in_(_closed_statuses),
   288→                or_(Bet.settled_at >= _today_start, Bet.closed_at >= _today_start),
   289→            )
   290→            .scalar()
   291→        ) or 0.0
   292→        daily_roi = (daily_pnl / total_stake_today) * 100 if total_stake_today > 0 else 0
   293→
   294→        # --- Sharpe Ratio & Max Drawdown ---
   295→        import math
   296→
   297→        sharpe_ratio = 0.0
   298→        max_drawdown_pct = 0.0
   299→        closed_bets = (
   300→            db.query(Bet.pnl, Bet.settled_at)
   301→            .filter(Bet.status.in_(_closed_statuses))
   302→            .order_by(Bet.settled_at.asc())
   303→            .all()
   304→        )
   305→        if len(closed_bets) > 1:
   306→            pnls = [float(b.pnl or 0.0) for b in closed_bets]
   307→            mean_pnl = sum(pnls) / len(pnls)
   308→            std_pnl = math.sqrt(sum((p - mean_pnl) ** 2 for p in pnls) / len(pnls))
   309→            sharpe_ratio = round(mean_pnl / std_pnl, 4) if std_pnl > 0 else 0.0
   310→
   311→            # Max Drawdown: simulate portfolio from initial capital
   312→            port_val = float(initial_capital)
   313→            peak = port_val
   314→            for b in closed_bets:
   315→                port_val += float(b.pnl or 0.0)
   316→                if port_val > peak:
   317→                    peak = port_val
   318→                dd = (peak - port_val) / peak * 100 if peak > 0 else 0
   319→                if dd > max_drawdown_pct:
   320→                    max_drawdown_pct = round(dd, 2)
   321→
   322→        return {
   323→            "is_running": state.is_running,
   324→            "locked": state.locked,
   325→            "portfolio": {
   326→                "initial": initial_capital,
   327→                "current": portfolio_current_value(initial_capital, realized_pnl_db, unrealized_pnl_db),
   328→                "daily_pnl": daily_pnl,
   329→                "daily_roi": daily_roi,
   330→                "unrealized_pnl": float(unrealized_pnl_db),
   331→                "realized_pnl": float(realized_pnl_db),
   332→                "total_pnl": total_pnl,
   333→                "total_roi": total_roi,
   334→                "exposure": float(exposure_db),
   335→                "max_exposure": round(
   336→                    max_exposure_cap(
   337→                        initial_capital,
   338→                        realized_before_today,
   339→                        state.config.TOTAL_EXPOSURE_PCT,
   340→                    ),
   341→                    2,
   342→                ),
   343→            },
   344→            "stats": {
   345→                "total_signals": total_signals_db,
   346→                "total_bets": total_bets_db,
   347→                "win_count": win_count,
   348→                "loss_count": loss_count,
   349→                "total_closed": win_count + loss_count,
   350→                "last_scan": state.last_scan.isoformat() if state.last_scan else None,
   351→            },
   352→            "limits": {
   353→                "max_bet_pct": state.config.MAX_BET_PCT * 100,
   354→                "max_exposure_pct": state.config.TOTAL_EXPOSURE_PCT * 100,
   355→                "daily_stop_loss_pct": state.config.DAILY_LOSS_LIMIT * 100,
   356→                "city_cap": state.config.CITY_CAP,
   357→            },
   358→            "metrics": {
   359→                "sharpe_ratio": sharpe_ratio,
   360→                "max_drawdown_pct": max_drawdown_pct,
   361→            },
   362→            "open_positions": open_positions,
   363→        }
   364→    except Exception as e:
   365→        return {"error": str(e)}
   366→    finally:
   367→        db.close()

   373→@app.get("/api/asi/weights")
   374→def get_asi_weights():
   375→    """Retrieve current evolved weights with model performance metrics."""
   376→    from database.models import ModelPerformance
   377→    from sqlalchemy import func
   378→
   379→    weights = load_weights()
   380→    if not weights:
   381→        weights = config.MODEL_WEIGHTS
   382→
   383→    # Get latest performance metrics for each model
   384→    db = get_db_session()
   385→    try:
   386→        # Subquery to get latest record per model
   387→        latest_perf = (
   388→            db.query(
   389→                ModelPerformance.model_name,
   390→                func.max(ModelPerformance.recorded_at).label("max_date"),
   391→            )
   392→            .group_by(ModelPerformance.model_name)
   393→            .subquery()
   394→        )
   395→
   396→        perf_records = (
   397→            db.query(ModelPerformance)
   398→            .join(
   399→                latest_perf,
   400→                (ModelPerformance.model_name == latest_perf.c.model_name)
   401→                & (ModelPerformance.recorded_at == latest_perf.c.max_date),
   402→            )
   403→            .all()
   404→        )
   405→
   406→        perf_map = {p.model_name: p for p in perf_records}
   407→    finally:
   408→        db.close()
   409→
   410→    # Return enriched weights with performance data
   411→    result = {}
   412→    for model, weight in weights.items():
   413→        perf = perf_map.get(model)
   414→        result[model] = {
   415→            "weight": weight,
   416→            "brier_score": perf.brier_score if perf else None,
   417→            "accuracy": perf.accuracy if perf else None,
   418→            "num_predictions": perf.num_predictions if perf else 0,
   419→            "last_updated": perf.recorded_at.isoformat() if perf and perf.recorded_at else None,
   420→        }
   421→    return result

   538→            .filter(
   539→                WeatherMarket.target_date >= now,
   540→                WeatherMarket.target_date <= upper,
   541→                WeatherMarket.status == "open",
   542→                ~WeatherMarket.id.in_([m["id"] for m in market_list]),
   543→            )
   544→            .limit(100)
   545→            .all()
   546→        )
   547→
   548→        calc = Calculator()
   549→        for m in markets:
   550→            if m.yes_price is None or m.threshold is None:
   551→                continue
   552→            current_price = float(m.yes_price)
   553→            model_prob = current_price
   554→            forecasts = (
   555→                db.query(WeatherForecast)
   556→                .filter(WeatherForecast.market_id == m.id)
   557→                .order_by(WeatherForecast.fetched_at.desc())
   558→                .limit(8)
   559→                .all()
   560→            )
   561→            if forecasts:
   562→                latest_vals = [f.predicted_value for f in forecasts]
   563→                days_ahead = max((m.target_date - now).days, 1)
   564→                model_prob = calc.estimate_probability(latest_vals, float(m.threshold), days_ahead)
   565→
   566→            market_list.append(
   567→                {
   568→                    "id": m.id,
   569→                    "city": m.city,
   570→                    "city_code": "",
   571→                    "date": m.target_date.isoformat() if m.target_date else None,
   572→                    "outcome_type": m.metric or "YES",
   573→                    "strike_temp": float(m.threshold),
   574→                    "current_yes_bid": current_price,
   575→                    "current_no_bid": m.no_price or (1 - current_price),
   576→                    "model_prob": model_prob,
   577→                    "edge": model_prob - current_price,
   578→                    "ev": safe_ev(model_prob, current_price),
   579→                    "status": "OPEN",
   580→                }
   581→            )
   582→
   583→        return {"markets": market_list, "count": len(market_list)}
   584→    except Exception as e:
   585→        logger.error("Markets API error: %s", e)
   586→        return {"error": str(e), "markets": []}

   716→@app.get("/api/history")
   717→def get_history():
   718→    """Get settled bet history with win/loss stats."""
   719→    db = get_db_session()
   720→    try:
   721→        from sqlalchemy import case, func
   722→
   723→        # True settlement stats: won+lost+settled+closed_early
   724→        # closed_early bets are real exits — their PnL is realized cash.
   725→        # Excluding them from stats gives a misleadingly small picture.
   726→        real_settled_statuses = ["settled", "won", "lost", "closed_early"]
   727→
   728→        stats_q = (
   729→            db.query(
   730→                func.count(Bet.id),
   731→                func.sum(case((Bet.pnl > 0, 1), else_=0)),
   732→                func.sum(case((Bet.pnl <= 0, 1), else_=0)),
   733→                func.coalesce(func.sum(Bet.amount), 0.0),
   734→                func.coalesce(func.sum(Bet.pnl), 0.0),
   735→                func.coalesce(func.sum(case((Bet.pnl > 0, Bet.pnl), else_=0.0)), 0.0),
   736→                func.coalesce(func.sum(case((Bet.pnl <= 0, func.abs(Bet.pnl)), else_=0.0)), 0.0),
   737→            )
   738→            .filter(Bet.status.in_(real_settled_statuses))
   739→            .one()
   740→        )
   741→        total_won = stats_q[1] or 0
   742→        total_lost = stats_q[2] or 0
   743→        total_stake_all = float(stats_q[3] or 0)
   744→        total_pnl_all = float(stats_q[4] or 0)
   745→        total_win_pnl = float(stats_q[5] or 0)
   746→        total_loss_pnl = float(stats_q[6] or 0)
   747→
   748→        # closed_early count and PnL (separate category)
   749→        ce_q = (
   750→            db.query(
   751→                func.count(Bet.id),
   752→                func.coalesce(func.sum(Bet.amount), 0.0),
   753→                func.coalesce(func.sum(Bet.pnl), 0.0),
   754→            )
   755→            .filter(Bet.status == "closed_early")
   756→            .one()
   757→        )
   758→        total_closed_early = ce_q[0] or 0
   759→        ce_pnl = float(ce_q[2] or 0)
   760→
   761→        # Average edge from settled bets (via Analysis join)
   762→        avg_edge_q = (
   763→            db.query(func.coalesce(func.avg(Analysis.edge), 0.0))
   764→            .join(Bet, Bet.analysis_id == Analysis.id)
   765→            .filter(Bet.status.in_(real_settled_statuses), Analysis.edge.isnot(None))
   766→            .scalar()
   767→        )
   768→        avg_edge = float(avg_edge_q or 0.0)
   769→
   770→        # History list: all settled + closed_early (most recent 300)
   771→        all_closed_statuses = ["settled", "won", "lost", "closed_early"]
   772→        # Use coalesce(settled_at, closed_at) for correct ordering
   773→        close_date = func.coalesce(Bet.settled_at, Bet.closed_at)
   774→        settled_bets = (
   775→            db.query(Bet)
   776→            .filter(Bet.status.in_(all_closed_statuses))
   777→            .order_by(close_date.desc(), Bet.placed_at.desc())
   778→            .limit(300)
   779→            .all()
   780→        )
   781→
   782→        # Batch-load analyses for edge data (N+1 fix)
   783→        bet_analysis_ids = [b.analysis_id for b in settled_bets if b.analysis_id]
   784→        bet_analyses = {}
   785→        if bet_analysis_ids:
   786→            for a in db.query(Analysis).filter(Analysis.id.in_(bet_analysis_ids)).all():
   787→                bet_analyses[a.id] = a
   788→
   789→        history = []
   790→        for bet in settled_bets:
   791→            pnl = bet.pnl or bet.realized_pnl or 0.0
   792→            stake = bet.amount or 0.0
   793→            roi = (pnl / stake * 100) if stake > 0 else 0.0
   794→            # Get actual edge from Analysis (net edge after slippage+fee)
   795→            analysis = bet_analyses.get(bet.analysis_id) if bet.analysis_id else None
   796→            edge_pct = round((analysis.edge or 0) * 100, 2) if analysis and analysis.edge else None
   797→            # Determine exit type code from status + close_reason
   798→            if bet.status == "closed_early" and bet.close_reason:
   799→                cr = bet.close_reason.lower()
   800→                if cr.startswith("take_profit"):
   801→                    exit_type = "TP"
   802→                elif cr.startswith("stop_loss"):
   803→                    exit_type = "SL"
   804→                elif cr.startswith("trailing_stop"):
   805→                    exit_type = "TS"
   806→                elif cr.startswith("time_decay"):
   807→                    exit_type = "TD"
   808→                else:
   809→                    exit_type = "OT"
   810→            else:
   811→                exit_type = "ST"  # settlement (Polymarket resolved)
   812→
   813→            history.append(
   814→                {
   815→                    "id": bet.id,
   816→                    "city": bet.city,
   817→                    "outcome": bet.side or "YES",
   818→                    "entry_price": bet.price,
   819→                    "exit_price": bet.current_price,  # actual fill price for early exits
   820→                    "stake_amount": stake,
   821→                    "realized_pnl": pnl,
   822→                    "roi": round(roi, 2),
   823→                    "edge": edge_pct,
   824→                    "result": "WIN" if pnl > 0 else "LOSS",
   825→                    "placed_at": bet.placed_at.isoformat() if bet.placed_at else None,
   826→                    "settled_at": (bet.settled_at.isoformat() if bet.settled_at else None),
   827→                    "closed_at": (bet.closed_at.isoformat() if bet.closed_at else None),
   828→                    "exit_type": exit_type,
   829→                }
   830→            )
   831→        win_rate = (total_won / (total_won + total_lost) * 100) if (total_won + total_lost) > 0 else 0
   832→        overall_roi = (total_pnl_all / total_stake_all * 100) if total_stake_all > 0 else 0.0
   833→        profit_factor = round(total_win_pnl / total_loss_pnl, 2) if total_loss_pnl > 0 else 0.0
   834→        return {
   835→            "history": history,
   836→            "stats": {
   837→                "total_won": total_won,
   838→                "total_lost": total_lost,
   839→                "total_closed_early": total_closed_early,
   840→                "closed_early_pnl": round(ce_pnl, 2),
   841→                "win_rate": round(win_rate, 2),
   842→                "overall_roi": round(overall_roi, 2),
   843→                "total_stake": round(total_stake_all, 2),
   844→                "total_pnl": round(total_pnl_all, 2),
   845→                "total_win_pnl": round(total_win_pnl, 2),
   846→                "total_loss_pnl": round(total_loss_pnl, 2),
   847→                "profit_factor": profit_factor,
   848→                "avg_edge": round(avg_edge * 100, 2) if avg_edge else 0.0,
   849→            },
   850→        }
   851→    finally:
   852→        db.close()

   855→@app.get("/api/equity-curve")
   856→def get_equity_curve():
   857→    """Daily equity curve from ALL settled + closed_early bets (no limit).
   858→
   859→    Returns [{date, pnl, count}] for every day that had at least one closure.
   860→    The frontend starts from INITIAL_PORTFOLIO and accumulates daily PnL.
   861→    """
   862→    from sqlalchemy import func
   863→
   864→    db = get_db_session()
   865→    try:
   866→        initial = state.config.INITIAL_PORTFOLIO
   867→        closed_statuses = ("won", "lost", "settled", "closed_early")
   868→        close_col = func.coalesce(Bet.settled_at, Bet.closed_at)
   869→        # SQLite: use date() function to extract calendar date from datetime string
   870→        day_expr = func.date(close_col)
   871→
   872→        # Group PnL by calendar date (UTC)
   873→        rows = (
   874→            db.query(
   875→                day_expr.label("day"),
   876→                func.sum(Bet.pnl).label("daily_pnl"),
   877→                func.count(Bet.id).label("cnt"),
   878→            )
   879→            .filter(Bet.status.in_(closed_statuses))
   880→            .group_by(day_expr)
   881→            .order_by(day_expr.asc())
   882→            .all()
   883→        )
   884→
   885→        points = []
   886→        running = initial
   887→        for row in rows:
   888→            # row.day is "YYYY-MM-DD" string from SQLite date()
   889→            day_str = row.day  # "2026-06-24"
   890→            pnl = float(row.daily_pnl or 0)
   891→            running += pnl
   892→            # Format: "24 Haz"
   893→            from datetime import datetime as dt
   894→
   895→            d = dt.strptime(day_str, "%Y-%m-%d")
   896→            label = f"{d.day} {d.strftime('%b')}"
   897→            points.append(
   898→                {
   899→                    "date": label,
   900→                    "value": round(running, 2),
   901→                    "pnl": round(pnl, 2),
   902→                    "count": row.cnt,
   903→                }
   904→            )
   905→
   906→        # Add today with current unrealized value
   907→        unrealized = (
   908→            db.query(func.coalesce(func.sum(Bet.unrealized_pnl), 0.0))
   909→            .filter(Bet.status.in_(OPEN_BET_STATUSES))
   910→            .scalar()
   911→        ) or 0.0
   912→        realized_now = running - initial  # all realized PnL accumulated
   913→        today_val = initial + realized_now + float(unrealized)
   914→        from datetime import datetime, timezone as tz
   915→
   916→        today = datetime.now(tz.utc).replace(tzinfo=None)
   917→        label = f"{today.day} {today.strftime('%b')}"
   918→        if points and points[-1]["date"] == label:
   919→            points[-1]["value"] = round(today_val, 2)
   920→        else:
   921→            points.append(
   922→                {
   923→                    "date": label,
   924→                    "value": round(today_val, 2),
   925→                    "pnl": 0,
   926→                    "count": 0,
   927→                }
   928→            )
   929→
   930→        return {"initial": initial, "points": points}
   931→    except Exception as e:
   932→        logger.error("Equity curve error: %s", e)
   933→        return {"error": str(e), "initial": config.INITIAL_PORTFOLIO, "points": []}
   934→    finally:
   935→        db.close()

  1008→
  1009→            # Calculate the actual debited amount — for ladder bets only
  1010→            # filled rungs were debited; for flat bets the full amount.
  1011→            from utils.accounting import credit_sale
  1012→
  1013→            ladder = _safe_parse_ladder(bet.ladder_data)
  1014→            if ladder:
  1015→                filled_amount = sum(float(rung.get("amount", 0)) for rung in ladder if rung.get("status") == "filled")

  1017→            else:
  1018→                refund_amount = float(bet.amount or 0)
  1019→
  1020→            credit_sale(db, refund_amount, f"cleanup_refund:bet_{bet.id}")
  1021→            cancelled += 1
