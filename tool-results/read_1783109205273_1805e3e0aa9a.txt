    98→    98→function fmtUsd(v: number) {
    99→    99→  const sign = v >= 0 ? "+" : "";
   100→   100→  return `${sign}$${Math.abs(v).toLocaleString("tr-TR", { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;
   101→   101→}
   102→   102→
   103→   103→// Turkish number format without sign (for prices)
   104→   104→function fmtPrice(v: number) {
   105→   105→  return v.toLocaleString("tr-TR", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
   106→   106→}
   107→   107→
   108→   108→// Turkish format for general numbers (percentages, ratios)
   109→   109→function fmtNum(v: number, decimals = 2) {
   110→   110→  return v.toLocaleString("tr-TR", { minimumFractionDigits: decimals, maximumFractionDigits: decimals });
   111→   111→}
   112→   112→
   113→   113→// Turkish format for integers (thousands separator, no decimals)
   114→   114→function fmtInt(v: number) {
   115→   115→  return v.toLocaleString("tr-TR", { maximumFractionDigits: 0 });
   116→   116→}
