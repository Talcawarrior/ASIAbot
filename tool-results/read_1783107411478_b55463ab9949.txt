     1→"""Layer 2: ASI-Evolve daily candidate generation loop.
     2→
     3→This is the "daily" layer in the 3-layer LLM stack:
     4→  Karpathy weekly (broad exploration, slow) ->
     5→  ASI-Evolve daily (UCB1-driven exploitation + experiment DB + FAISS cognition) ->
     6→  SIA hourly (LLM-driven harness patches, fast weight tuning).
     7→
     8→Ported from the design of GAIR-NLP/ASI-Evolve: a 3-agent loop with
     9→Researcher / Engineer / Analyzer roles, an experiment DB (SQLite) that
    10→records every candidate's parent + metrics, and a FAISS cognition store
    11→that retrieves similar past experiments to inform new proposals.
    12→
    13→Key features:
    14→
    15→  - **UCB1 parent selection**: instead of always mutating the current best,
    16→    pick the parent using Upper Confidence Bound 1 (Sutton & Barto §2.7):
    17→        UCB1(n) = mean_roi(n) + c * sqrt(2 * ln(N) / n_visits(n)
    18→    This balances exploitation (high-ROI parents) with exploration
    19→    (under-visited parents), producing 50-200 candidates per run as the
    20→    ASI-Evolve paper suggests.
    21→
    22→  - **Experiment DB**: every candidate is persisted with its parent ID,
    23→    hypothesis text, params, and OOS metrics. Future runs query this DB
    24→    for both UCB1 stats and FAISS retrieval.
    25→
    26→  - **FAISS cognition store**: each candidate's hypothesis text is embedded
    27→    (sentence-transformers if installed, else hash-based pseudo-embedding)
    28→    and indexed. When the Researcher proposes a new candidate, the top-K
    29→    most similar past experiments are retrieved and fed as context.
    30→
    31→  - **Real evaluation**: each candidate is evaluated using the same
    32→    walk-forward OOS Brier dataset as Layer 1 (no synthetic noise).
    33→
    34→  - **No LLM hard-dependency**: the Engineer agent uses the LLM if
    35→    ``ZAI_API_KEY`` is set (via the shared ``asi_engine.llm_client``
    36→    helper), otherwise falls back to the Layer 1 mutation ladder. CI
    37→    runs without network.
    38→"""

   115→# ---------------------------------------------------------------------------
   116→# FAISS cognition store (or fallback)
   117→# ---------------------------------------------------------------------------
   118→
   119→
   120→class CognitionStore:
   121→    """Vector store of past experiment hypotheses for retrieval.
   122→
   123→    Uses FAISS + sentence-transformers if installed; otherwise falls back to
   124→    a hash-based pseudo-embedding that supports the same retrieve() API
   125→    (with worse recall but no extra deps). The fallback is sufficient for CI.
   126→    """
   127→
   128→    def __init__(self, dim: int = 384):
   129→        self.dim = dim
   130→        self._ids: list[int] = []
   131→        self._texts: list[str] = []
   132→        self._embeddings: list[list[float]] = []
   133→        self._faiss_index = None
   134→        self._embedder = None
   135→        self._init_backend()
   136→
   137→    def _init_backend(self):
   138→        try:
   139→            import faiss  # type: ignore
   140→            import sentence_transformers  # type: ignore
   141→
   142→            self._embedder = sentence_transformers.SentenceTransformer("sentence-transformers/all-MiniLM-L6-v2")
   143→            self._faiss_index = faiss.IndexFlatIP(self.dim)
   144→            logger.info("CognitionStore: FAISS + sentence-transformers loaded")
   145→        except ImportError:
   146→            logger.info(
   147→                "CognitionStore: faiss/sentence-transformers not installed — using hash-based fallback embeddings"
   148→            )
   149→
   150→    def _embed(self, text: str) -> list[float]:
   151→        if self._embedder is not None:
   152→            vec = self._embedder.encode([text], normalize_embeddings=True)[0]
   153→            return vec.tolist()
   154→        # Fallback: deterministic hash-based pseudo-embedding
   155→        rng = random.Random(hash(text) & 0xFFFFFFFF)
   156→        return [rng.gauss(0, 1) for _ in range(self.dim)]
   157→
   158→    def add(self, exp_id: int, text: str) -> None:
   159→        vec = self._embed(text)
   160→        # Normalise hash-fallback to unit length for cosine similarity
   161→        norm = math.sqrt(sum(v * v for v in vec))
   162→        if norm > 0:
   163→            vec = [v / norm for v in vec]
   164→        self._ids.append(exp_id)
   165→        self._texts.append(text)
   166→        self._embeddings.append(vec)
   167→        if self._faiss_index is not None:
   168→            import numpy as np  # type: ignore
   169→
   170→            self._faiss_index.add(np.array([vec], dtype="float32"))
   171→
   172→    def retrieve(self, query: str, k: int = 3) -> list[dict[str, Any]]:
   173→        """Return top-K similar past experiments.
   174→
   175→        Each result is {id, text, similarity}.
   176→        """
   177→        if not self._embeddings:
   178→            return []
   179→        q = self._embed(query)
   180→        norm = math.sqrt(sum(v * v for v in q))
   181→        if norm > 0:
   182→            q = [v / norm for v in q]
   183→
   184→        if self._faiss_index is not None:
   185→            import numpy as np  # type: ignore
   186→
   187→            d, idx_arr = self._faiss_index.search(np.array([q], dtype="float32"), min(k, len(self._ids)))
   188→            results = []
   189→            for sim, idx in zip(d[0].tolist(), idx_arr[0].tolist()):
   190→                if idx < 0:
   191→                    continue
   192→                results.append(
   193→                    {
   194→                        "id": self._ids[idx],
   195→                        "text": self._texts[idx],
   196→                        "similarity": float(sim),
   197→                    }
   198→                )
   199→            return results
   200→
   201→        # Fallback: cosine similarity against all stored vectors
   202→        sims = []
   203→        for i, v in enumerate(self._embeddings):
   204→            s = sum(a * b for a, b in zip(q, v))
   205→            sims.append((s, i))
   206→        sims.sort(reverse=True)
   207→        return [{"id": self._ids[i], "text": self._texts[i], "similarity": float(s)} for s, i in sims[:k]]

   210→# ---------------------------------------------------------------------------
   211→# UCB1 parent selection
   212→# ---------------------------------------------------------------------------
   213→
   214→
   215→def ucb1_select_parent(
   216→    conn: sqlite3.Connection,
   217→    exploration_c: float = 1.41,
   218→    min_visits: int = 1,
   219→) -> int | None:
   220→    """Pick the parent ID for the next candidate using UCB1.
   221→
   222→    UCB1(n) = mean_roi(n) + c * sqrt(2 * ln(N_total) / n_visits(n))
   223→
   224→    Returns None if there are no experiments yet (caller should use the
   225→    uniform prior as parent).
   226→    """
   227→    cur = conn.cursor()
   228→    cur.execute("SELECT COUNT(*) FROM experiments")
   229→    n = cur.fetchone()[0]
   230→    if n == 0:
   231→        return None
   232→
   233→    # For each distinct parent_id, compute mean ROI and visit count
   234→    cur.execute(
   235→        """
   236→        SELECT parent_id, COUNT(*) as visits, AVG(roi_pct) as mean_roi
   237→        FROM experiments
   238→        WHERE parent_id IS NOT NULL
   239→        GROUP BY parent_id
   240→        """
   241→    )
   242→    rows = cur.fetchall()
   243→    if not rows:
   244→        return None
   245→
   246→    ln_n = math.log(max(n, 1))
   247→    best_score = -1e18
   248→    best_parent: int | None = None
   249→    for parent_id, visits, mean_roi in rows:
   250→        if visits < min_visits:
   251→            # Force exploration of under-visited parents
   252→            return parent_id
   253→        ucb = mean_roi + exploration_c * math.sqrt(ln_n / visits)
   254→        if ucb > best_score:
   255→            best_score = ucb
   256→            best_parent = parent_id
   257→
   258→    return best_parent

   261→def get_parent_hypothesis(conn: sqlite3.Connection, parent_id: int) -> Hypothesis | None:
   262→    """Load a parent hypothesis from the experiment DB."""
   263→    cur = conn.cursor()
   264→    cur.execute(
   265→        "SELECT hypothesis_json FROM experiments WHERE id = ?",
   266→        (parent_id,),
   267→    )
   268→    row = cur.fetchone()
   269→    if not row:
   270→        return None
   271→    try:
   272→        data = json.loads(row[0])
   273→        clean = {k: v for k, v in data.items() if k in Hypothesis.__dataclass_fields__}
   274→        return Hypothesis(**clean)
   275→    except Exception as e:
   276→        logger.warning("Could not load parent %d: %s", parent_id, e)
   277→        return None

   284→
   285→@dataclass
   286→class Candidate:
   287→    """A proposed hypothesis + its origin metadata."""
   288→
   289→    hypothesis: Hypothesis
   290→    parent_id: int | None
   291→    source: str  # 'mutation_ladder' / 'llm' / 'crossover' / 'seed'
   292→    round_num: int
   293→    candidate_idx: int
   294→
   295→
   296→class ResearcherAgent:
   297→    """Selects the parent and decides *what kind* of mutation to propose."""
   298→
   299→    def __init__(self, conn: sqlite3.Connection, cognition: CognitionStore):
   300→        self.conn = conn
   301→        self.cognition = cognition
   302→
   303→    def select_parent(self) -> tuple[Hypothesis, int | None]:
   304→        """Use UCB1 to pick a parent. Returns (parent_hyp, parent_id)."""
   305→        parent_id = ucb1_select_parent(self.conn)
   306→        if parent_id is None:
   307→            # No experiments yet — use the Layer 1 best or uniform prior
   308→            from asi_engine.karpathy_weekly import _load_best
   309→
   310→            parent = _load_best() or Hypothesis(
   311→                description="Uniform prior (no parent)",
   312→                model_weights=_uniform_weights(),
   313→                min_edge=0.05,
   314→                kelly_fraction=0.15,
   315→                max_bet_pct=0.05,
   316→            )
   317→            return parent, None
   318→        parent = get_parent_hypothesis(self.conn, parent_id)
   319→        if parent is None:
   320→            # Parent ID exists but hypothesis couldn't be loaded — fall back
   321→            from asi_engine.karpathy_weekly import _load_best
   322→
   323→            parent = _load_best() or Hypothesis(
   324→                description="Fallback",
   325→                model_weights=_uniform_weights(),
   326→                min_edge=0.05,
   327→                kelly_fraction=0.15,
   328→                max_bet_pct=0.05,
   329→            )
   330→            return parent, None
   331→        return parent, parent_id
   332→
   333→    def retrieve_context(self, query: str) -> list[dict[str, Any]]:
   334→        """Retrieve top-K similar past experiments from cognition store."""
   335→        return self.cognition.retrieve(query, k=3)

   338→class EngineerAgent:
   339→    """Generates the actual candidate hypothesis (the 'code patch').
   340→
   341→    Uses the LLM if available; otherwise falls back to the Layer 1
   342→    mutation ladder.
   343→    """
   344→
   345→    def __init__(self, use_llm: bool = True):
   346→        self.use_llm = use_llm
   347→
   348→    def propose(
   349→        self,
   350→        parent: Hypothesis,
   351→        round_num: int,
   352→        candidate_idx: int,
   353→        context: list[dict[str, Any]] | None = None,
   354→    ) -> Hypothesis:
   355→        # Try LLM first if enabled
   356→        if self.use_llm:
   357→            ctx_payload = {
   358→                "parent": parent.to_dict(),
   359→                "similar_experiments": context or [],
   360→            }
   361→            hyp = llm_propose_hypothesis(parent, ctx_payload)
   362→            if hyp is not None:

   365→        # Fallback: deterministic mutation ladder from Layer 1
   366→        return karpathy_generate(round_num + candidate_idx, parent=parent)

   369→class AnalyzerAgent:
   370→    """Evaluates a candidate and persists results to the experiment DB."""
   371→
   372→    def __init__(self, conn: sqlite3.Connection, cognition: CognitionStore):
   373→        self.conn = conn

   388→        # Persist to experiment DB

   390→        cur.execute(
   391→            """
   392→            INSERT INTO experiments (
   393→                parent_id, round, candidate_idx, description, source,
   394→                hypothesis_json, stats_json, sharpe, roi_pct, brier_score,
   395→                total_trades, accepted, created_at
   396→            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
   397→            """,
   398→            (
   399→                candidate.parent_id,
   400→                candidate.round_num,
   401→                candidate.candidate_idx,
   402→                candidate.hypothesis.description,
   403→                candidate.source,
   404→                json.dumps(candidate.hypothesis.to_dict(), sort_keys=True),
   405→                json.dumps(mean_stats, sort_keys=True),
   406→                mean_stats["sharpe"],
   407→                mean_stats["roi_pct"],
   408→                mean_stats["brier_score"],
   409→                mean_stats["total_trades"],
   410→                0,  # accepted flag set later
   411→                datetime.now(UTC).isoformat(),
   412→            ),
   413→        )
   414→        self.conn.commit()
   415→        exp_id = cur.lastrowid
   416→
   417→        # Add to cognition store for future retrieval
   418→        self.cognition.add(exp_id, candidate.hypothesis.description)
   419→
   420→        mean_stats["exp_id"] = exp_id
   421→        return mean_stats

   424→# ---------------------------------------------------------------------------
   425→# Crossover operator (for diversity)
   426→# ---------------------------------------------------------------------------
   427→
   428→
   429→def crossover(h1: Hypothesis, h2: Hypothesis) -> Hypothesis:
   430→    """Uniform crossover between two hypotheses' weights.
   431→
   432→    Takes each model's weight from h1 or h2 with 50% probability, then
   433→    re-normalises. Other params are taken from the better parent (caller's
   434→    responsibility to choose).
   435→    """
   436→    new_weights = {}
   437→    for m in DEFAULT_MODELS:
   438→        if random.random() < 0.5:
   439→            new_weights[m] = h1.model_weights.get(m, 0.125)
   440→        else:
   441→            new_weights[m] = h2.model_weights.get(m, 0.125)
   442→    new_weights = _normalise(new_weights)
   443→
   444→    return Hypothesis(
   445→        description=f"Crossover({h1.description[:30]} × {h2.description[:30]})",
   446→        model_weights=new_weights,
   447→        min_edge=(h1.min_edge + h2.min_edge) / 2,
   448→        kelly_fraction=(h1.kelly_fraction + h2.kelly_fraction) / 2,
   449→        max_bet_pct=max(h1.max_bet_pct, h2.max_bet_pct),
   450→        tail_filter_enabled=h1.tail_filter_enabled or h2.tail_filter_enabled,
   451→        tail_filter_threshold_high=h1.tail_filter_threshold_high,
   452→        tail_filter_threshold_low=h1.tail_filter_threshold_low,
   453→        tail_filter_correction_high=h1.tail_filter_correction_high,
   454→        tail_filter_correction_low=h1.tail_filter_correction_low,
   455→        source="crossover",
   456→    )

   460→# Main loop

   463→
   464→def _mean_stats(per_split: list[dict[str, float]]) -> dict[str, float]:
   465→    """Mean across splits, equal-weighted."""
   466→    if not per_split:
   467→        return {
   468→            "sharpe": 0.0,
   469→            "roi_pct": 0.0,
   470→            "win_rate": 0.0,
   471→            "total_trades": 0,
   472→            "brier_score": 0.25,
   473→            "total_pnl": 0.0,
   474→            "total_staked": 0.0,
   475→        }
   476→    keys = ["sharpe", "roi_pct", "win_rate", "brier_score", "total_pnl", "total_staked"]
   477→    out = {k: sum(s.get(k, 0.0) for s in per_split) / len(per_split) for k in keys}
   478→    out["total_trades"] = int(sum(s.get("total_trades", 0) for s in per_split))
   479→    out["sharpe"] = round(out["sharpe"], 4)
   480→    out["roi_pct"] = round(out["roi_pct"], 4)
   481→    out["brier_score"] = round(out["brier_score"], 4)
   482→    out["win_rate"] = round(out["win_rate"], 4)
   483→    return out

   486→def _load_best(
   487→    conn: sqlite3.Connection,
   488→) -> tuple[Hypothesis | None, dict[str, float] | None]:
   489→    """Load the current best hypothesis from the experiment DB."""
   490→    cur = conn.cursor()
   491→    cur.execute(
   492→        "SELECT id, hypothesis_json, stats_json FROM experiments WHERE accepted = 1 ORDER BY sharpe DESC LIMIT 1"
   493→    )
   494→    row = cur.fetchone()
   495→    if not row:
   496→        return None, None
   497→    try:
   498→        data = json.loads(row[1])
   499→        clean = {k: v for k, v in data.items() if k in Hypothesis.__dataclass_fields__}
   500→        hyp = Hypothesis(**clean)
   501→        stats = json.loads(row[2])
   502→        return hyp, stats
   503→    except Exception as e:
   504→        logger.warning("Could not load best: %s", e)
   505→        return None, None

   543→def run_asi_evolve_daily(
   544→    n_candidates: int = 20,
   545→    use_llm: bool = False,
   546→    crossover_rate: float = 0.2,
   547→    seed: int = 42,
   548→) -> dict[str, Any]:
   549→    """Run one ASI-Evolve daily iteration.
   550→
   551→    Args:
   552→        n_candidates: number of candidates to generate + evaluate (paper
   553→            suggests 50-200; we default to 20 for CI speed).
   554→        use_llm: if True, Engineer uses LLM proposals when available.
   555→        crossover_rate: fraction of candidates produced via crossover
   556→            rather than mutation.
   557→        seed: RNG seed.
   558→
   559→    Returns a summary dict.
   560→    """
   561→    random.seed(seed)
   562→    conn = _get_db()
   563→    cognition = CognitionStore()
   564→
   565→    # Pre-populate cognition store with all past experiments
   566→    cur = conn.cursor()
   567→    cur.execute("SELECT id, description FROM experiments ORDER BY id")
   568→    for exp_id, desc in cur.fetchall():
   569→        cognition.add(exp_id, desc)
   570→
   571→    # 1. Pull unified Brier dataset + splits
   572→    ds = UnifiedDatastore()
   573→    try:
   574→        brier_df = ds.build_brier_dataset()
   575→    except Exception as e:
   576→        logger.warning(
   577→            "build_brier_dataset() raised %s — run polymarket_ingest with the "
   578→            "weather market parser first. Returning early.",
   579→            e,
   580→        )
   581→        return {
   582→            "error": "brier_dataset_unavailable",
   583→            "detail": str(e),
   584→            "candidates_run": 0,
   585→        }
   586→
   587→    if brier_df is None or brier_df.empty:
   588→        logger.error("Brier dataset is empty")
   589→        return {"error": "empty_brier_dataset", "candidates_run": 0}
   590→
   591→    # Add per-model prob columns (tries real forecast join, falls back to synthetic)
   592→    from asi_engine.karpathy_weekly import add_per_model_probabilities
   593→
   594→    brier_df = add_per_model_probabilities(brier_df, ds=ds)
   595→
   596→    splits = ds.build_walk_forward_splits()
   597→    if not splits:
   598→        splits = [
   599→            {
   600→                "split_n": 1,
   601→                "test_indices": brier_df.index.tolist(),
   602→                "train_indices": [],
   603→            }
   604→        ]
   605→
   606→    # 2. Init agents
   607→    researcher = ResearcherAgent(conn, cognition)
   608→    engineer = EngineerAgent(use_llm=use_llm)
   609→    analyzer = AnalyzerAgent(conn, cognition)
   610→
   611→    # 3. Load current best
   612→    best_hyp, best_stats = _load_best(conn)
   613→    if best_hyp is None:
   614→        # Seed with the Karpathy best (if any) or uniform prior
   615→        from asi_engine.karpathy_weekly import _load_best as load_karpathy_best
   616→
   617→        best_hyp = load_karpathy_best() or Hypothesis(
   618→            description="Uniform prior (seed)",
   619→            model_weights=_uniform_weights(),
   620→            min_edge=0.05,
   621→            kelly_fraction=0.15,
   622→            max_bet_pct=0.05,
   623→        )
   624→        best_stats = {
   625→            "sharpe": -1e9,
   626→            "roi_pct": -1e9,
   627→            "brier_score": 1.0,
   628→            "win_rate": 0.0,
   629→            "total_trades": 0,
   630→            "total_pnl": 0.0,
   631→            "total_staked": 0.0,
   632→        }
   633→        # Seed the DB so future UCB1 has a parent to point to
   634→        seed_candidate = Candidate(
   635→            hypothesis=best_hyp,
   636→            parent_id=None,
   637→            source="seed",
   638→            round_num=0,
   639→            candidate_idx=0,
   640→        )
   641→        best_stats = analyzer.evaluate_and_store(seed_candidate, brier_df, splits)
   642→        # Mark as accepted
   643→        cur.execute(
   644→            "UPDATE experiments SET accepted = 1 WHERE id = ?",
   645→            (best_stats["exp_id"],),
   646→        )
   647→        conn.commit()
   648→        _save_best_metadata(best_hyp, best_stats)
   649→
   650→    # 4. Generate + evaluate candidates
   651→    round_num = 1 + (cur.execute("SELECT COALESCE(MAX(round), 0) FROM experiments").fetchone()[0])
   652→    accepted_count = 0
   653→
   654→    for c in range(n_candidates):
   655→        # Pick parent via UCB1
   656→        parent_hyp, parent_id = researcher.select_parent()
   657→
   658→        # Decide: crossover or mutation?
   659→        if random.random() < crossover_rate and len(cognition._ids) >= 2:
   660→            # Pick a second parent for crossover
   661→            second_id = random.choice([i for i in cognition._ids if i != parent_id])
   662→            second_hyp = get_parent_hypothesis(conn, second_id) or parent_hyp
   663→            new_hyp = crossover(parent_hyp, second_hyp)
   664→            source = "crossover"
   665→        else:
   666→            # Mutation
   667→            context = researcher.retrieve_context(parent_hyp.description)
   668→            new_hyp = engineer.propose(parent_hyp, round_num=round_num, candidate_idx=c, context=context)
   669→            source = new_hyp.source
   670→
   671→        candidate = Candidate(
   672→            hypothesis=new_hyp,
   673→            parent_id=parent_id,
   674→            source=source,
   675→            round_num=round_num,
   676→            candidate_idx=c,
   677→        )
   678→
   679→        try:
   680→            stats = analyzer.evaluate_and_store(candidate, brier_df, splits)
   681→        except Exception as e:
   682→            logger.exception("Candidate %d evaluation failed: %s", c, e)
   683→            continue
   684→
   685→        improved = (
   686→            stats["sharpe"] > best_stats.get("sharpe", -1e9)
   687→            and stats["brier_score"] <= best_stats.get("brier_score", 1.0) * 1.10
   688→            and stats["total_trades"] >= 5
   689→        )
   690→
   691→        if improved:
   692→            logger.info(
   693→                "  [%d/%d] ✓ ACCEPTED (sharpe %.3f > %.3f): %s",
   694→                c + 1,
   695→                n_candidates,
   696→                stats["sharpe"],
   697→                best_stats.get("sharpe", 0.0),
   698→                new_hyp.description,
   699→            )
   700→            best_hyp = new_hyp
   701→            best_stats = stats
   702→            cur.execute(
   703→                "UPDATE experiments SET accepted = 1 WHERE id = ?",
   704→                (stats["exp_id"],),
   705→            )
   706→            conn.commit()
   707→            _save_best_metadata(new_hyp, stats)
   708→            _append_results_tsv(round_num, c, new_hyp, stats, "keep")
   709→            accepted_count += 1
   710→        else:
   711→            logger.info(
   712→                "  [%d/%d] ✗ reject (sharpe %.3f ≤ %.3f)",
   713→                c + 1,
   714→                n_candidates,
   715→                stats["sharpe"],
   716→                best_stats.get("sharpe", 0.0),
   717→            )
   718→            _append_results_tsv(round_num, c, new_hyp, stats, "reject")
   719→
   720→        # LLM rate-limit friendliness
   721→        if use_llm and (c + 1) % 5 == 0:
   722→            time.sleep(1.0)
   723→
   724→    cur.execute("SELECT COUNT(*) FROM experiments")
   725→    total_experiments = cur.fetchone()[0]
   726→    conn.close()
   727→
   728→    logger.info(
   729→        "ASI-Evolve round %d done. Candidates=%d, Accepted=%d, Total DB size=%d",
   730→        round_num,
   731→        n_candidates,
   732→        accepted_count,
   733→        total_experiments,
   734→    )
   735→
   736→    return {
   737→        "round": round_num,
   738→        "candidates_run": n_candidates,
   739→        "accepted": accepted_count,
   740→        "total_experiments": total_experiments,
   741→        "best_hypothesis": best_hyp.to_dict(),
   742→        "best_stats": best_stats,
   743→        "n_splits": len(splits),
   744→    }
   745→

   758→    )
   759→
   760→    parser = argparse.ArgumentParser(description="ASI-Evolve daily candidate generation loop")
   761→    parser.add_argument("--candidates", type=int, default=20, help="Number of candidates per run")
   762→    parser.add_argument("--llm", action="store_true", help="Use LLM for hypothesis generation")
   763→    parser.add_argument("--crossover", type=float, default=0.2, help="Crossover probability")
