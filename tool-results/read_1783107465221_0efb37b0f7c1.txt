   299‚Üí   299‚Üí        closed_bets = (

   307‚Üí   307‚Üí            mean_pnl = sum(pnls) / len(pnls)
   308‚Üí   308‚Üí            std_pnl = math.sqrt(sum((p - mean_pnl) ** 2 for p in pnls) / len(pnls))
   309‚Üí   309‚Üí            sharpe_ratio = round(mean_pnl / std_pnl, 4) if std_pnl > 0 else 0.0
   310‚Üí   310‚Üí
   311‚Üí   311‚Üí            # Max Drawdown: simulate portfolio from initial capital

   316‚Üí   316‚Üí                if port_val > peak:

   318‚Üí   318‚Üí                dd = (peak - port_val) / peak * 100 if peak > 0 else 0
   319‚Üí   319‚Üí                if dd > max_drawdown_pct:
   320‚Üí   320‚Üí                    max_drawdown_pct = round(dd, 2)
   321‚Üí   321‚Üí
   322‚Üí   322‚Üí        return {

   716‚Üí   716‚Üí@app.get("/api/history")
   717‚Üí   717‚Üídef get_history():

   720‚Üí   720‚Üí    try:
   721‚Üí   721‚Üí        from sqlalchemy import case, func
   722‚Üí   722‚Üí
   723‚Üí   723‚Üí        # True settlement stats: won+lost+settled+closed_early
   724‚Üí   724‚Üí        # closed_early bets are real exits ‚Äî their PnL is realized cash.

   731‚Üí   731‚Üí                func.sum(case((Bet.pnl > 0, 1), else_=0)),
   732‚Üí   732‚Üí                func.sum(case((Bet.pnl <= 0, 1), else_=0)),
   733‚Üí   733‚Üí                func.coalesce(func.sum(Bet.amount), 0.0),

   736‚Üí   736‚Üí                func.coalesce(func.sum(case((Bet.pnl <= 0, func.abs(Bet.pnl)), else_=0.0)), 0.0),
   737‚Üí   737‚Üí            )
   738‚Üí   738‚Üí            .filter(Bet.status.in_(real_settled_statuses))

   797‚Üí   797‚Üí            # Determine exit type code from status + close_reason
   798‚Üí   798‚Üí            if bet.status == "closed_early" and bet.close_reason:

   856‚Üí   856‚Üídef get_equity_curve():
   857‚Üí   857‚Üí    """Daily equity curve from ALL settled + closed_early bets (no limit).
   858‚Üí   858‚Üí
   859‚Üí   859‚Üí    Returns [{date, pnl, count}] for every day that had at least one closure.
   860‚Üí   860‚Üí    The frontend starts from INITIAL_PORTFOLIO and accumulates daily PnL.
   861‚Üí   861‚Üí    """
   862‚Üí   862‚Üí    from sqlalchemy import func
   863‚Üí   863‚Üí
   864‚Üí   864‚Üí    db = get_db_session()
   865‚Üí   865‚Üí    try:
   866‚Üí   866‚Üí        initial = state.config.INITIAL_PORTFOLIO
   867‚Üí   867‚Üí        closed_statuses = ("won", "lost", "settled", "closed_early")
   868‚Üí   868‚Üí        close_col = func.coalesce(Bet.settled_at, Bet.closed_at)
   869‚Üí   869‚Üí        # SQLite: use date() function to extract calendar date from datetime string
   870‚Üí   870‚Üí        day_expr = func.date(close_col)
   871‚Üí   871‚Üí
   872‚Üí   872‚Üí        # Group PnL by calendar date (UTC)
   873‚Üí   873‚Üí        rows = (
   874‚Üí   874‚Üí            db.query(
   875‚Üí   875‚Üí                day_expr.label("day"),
   876‚Üí   876‚Üí                func.sum(Bet.pnl).label("daily_pnl"),
   877‚Üí   877‚Üí                func.count(Bet.id).label("cnt"),
   878‚Üí   878‚Üí            )
   879‚Üí   879‚Üí            .filter(Bet.status.in_(closed_statuses))
   880‚Üí   880‚Üí            .group_by(day_expr)
   881‚Üí   881‚Üí            .order_by(day_expr.asc())
   882‚Üí   882‚Üí            .all()
   883‚Üí   883‚Üí        )
   884‚Üí   884‚Üí
   885‚Üí   885‚Üí        points = []
   886‚Üí   886‚Üí        running = initial
   887‚Üí   887‚Üí        for row in rows:
   888‚Üí   888‚Üí            # row.day is "YYYY-MM-DD" string from SQLite date()
   889‚Üí   889‚Üí            day_str = row.day  # "2026-06-24"
   890‚Üí   890‚Üí            pnl = float(row.daily_pnl or 0)
   891‚Üí   891‚Üí            running += pnl
   892‚Üí   892‚Üí            # Format: "24 Haz"
   893‚Üí   893‚Üí            from datetime import datetime as dt
   894‚Üí   894‚Üí
   895‚Üí   895‚Üí            d = dt.strptime(day_str, "%Y-%m-%d")
   896‚Üí   896‚Üí            label = f"{d.day} {d.strftime('%b')}"
   897‚Üí   897‚Üí            points.append(
   898‚Üí   898‚Üí                {
   899‚Üí   899‚Üí                    "date": label,
   900‚Üí   900‚Üí                    "value": round(running, 2),
   901‚Üí   901‚Üí                    "pnl": round(pnl, 2),
   902‚Üí   902‚Üí                    "count": row.cnt,
   903‚Üí   903‚Üí                }
   904‚Üí   904‚Üí            )
