     1→"""Layer 3: SIA hourly weight + harness update loop.
     2→
     3→This is the "hourly" layer in the 3-layer LLM stack — the fastest,
     4→narrowest layer. Ported from the design of hexo-ai/sia: a 3-agent
     5→Meta/Target/Feedback loop with two update tracks:
     6→
     7→  1. **Weight update**: nudges the model_weights dict (same surface as
     8→     Layer 1's mutation ladder, but smaller, faster steps). This is the
     9→     SIA "weight update" branch — it runs even without an LLM.
    10→
    11→  2. **Harness update**: asks the LLM to propose a *code patch* to the
    12→     harness (the function that turns per-model forecasts into a YES
    13→     probability). This is the SIA "harness update" branch — it only
    14→     runs when an LLM is available, and the patch must pass a syntax
    15→     check + smoke-eval before being accepted.
    16→
    17→Compared to Layer 2 (ASI-Evolve, daily, 50-200 candidates, UCB1):
    18→  - Layer 3 runs hourly (or on-demand).
    19→  - Layer 3 generates 1-3 candidates per run (vs 50-200).
    20→  - Layer 3 may modify the *harness code itself* (Layer 2 only mutates
    21→    weights/params).
    22→  - Layer 3 always starts from the Layer 2 best, never from scratch.
    23→
    24→The harness being patched is `asi_engine/sia_harness.py` — a small
    25→file containing one function `predict_yes_probability(forecasts,
    26→weights, days_ahead) -> float`. The LLM is shown the current source,
    27→asked for a diff, the diff is applied, the patched file is imported
    28→in a subprocess (sandbox) and smoke-evaluated against a small OOS
    29→window. Only if the patched harness beats the incumbent does it get
    30→persisted.
    31→
    32→If no LLM is available, only the weight-update branch runs (no harness
    33→patches) — this keeps the layer useful in CI.
    34→"""

    77→DEFAULT_HARNESS_SRC = '''"""SIA harness — the function that turns per-model forecasts into a YES prob.
    78→
    79→This file is the target of Layer 3 harness updates. The LLM may propose
    80→patches to `predict_yes_probability`. Each patch must pass a syntax
    81→check + smoke-eval before being accepted.

    84→from __future__ import annotations
    85→
    86→import math
    87→
    88→
    89→def predict_yes_probability(
    90→    forecasts: dict[str, float],
    91→    weights: dict[str, float],
    92→    threshold: float,
    93→    days_ahead: int = 1,
    94→) -> float:
    95→    """Compute P(YES) from per-model temperature forecasts.
    96→
    97→    Default implementation: weighted mean of forecasts -> z-score vs
    98→    threshold -> Normal CDF.
    99→
   100→    Args:
   101→        forecasts: dict of {model_name: forecasted_temperature}
   102→        weights: dict of {model_name: weight} (must sum to 1.0)
   103→        threshold: market threshold temperature
   104→        days_ahead: forecast horizon (1-7)
   105→
   106→    Returns:
   107→        P(YES) in [0.01, 0.99]
   108→    """
   109→    if not forecasts:
   110→        return 0.5
   111→
   112→    wsum = sum(weights.get(m, 0.0) for m in forecasts)
   113→    if wsum <= 0:
   114→        return 0.5
   115→
   116→    weighted_mean = sum(weights.get(m, 0.0) * f for m, f in forecasts.items()) / wsum
   117→    variance = sum(
   118→        weights.get(m, 0.0) * (f - weighted_mean) ** 2
   119→        for m, f in forecasts.items()
   120→    ) / wsum
   121→    std = math.sqrt(variance) if variance > 0 else 1.0
   122→    effective_std = std * math.sqrt(max(days_ahead, 1))
   123→
   124→    z = (weighted_mean - threshold) / (effective_std + 1e-5)
   125→    p = 0.5 * (1.0 + math.erf(z / math.sqrt(2.0)))
   126→    return min(max(p, 0.01), 0.99)
   127→'''

   162→@dataclass
   163→class SIAState:
   164→    """State carried between agents in one SIA cycle."""
   165→
   166→    parent_hypothesis: Hypothesis
   167→    parent_stats: dict[str, float]
   168→    candidate_hypothesis: Hypothesis | None = None
   169→    candidate_stats: dict[str, float] | None = None

   175→
   176→class MetaAgent:
   177→    """Orchestrates the cycle: picks what to mutate (weights vs harness).
   178→
   179→    The Meta agent looks at the parent's recent stats:
   180→      - If Brier is high (>0.25), the harness is the bottleneck → suggest
   181→        harness patch (if LLM available).
   182→      - If Brier is OK but Sharpe is low, weights are the bottleneck →
   183→        suggest weight mutation.
   184→      - If both are bad, try both.
   185→    """
   186→
   187→    def __init__(self, use_llm: bool):
   188→        self.use_llm = use_llm
   189→
   190→    def decide(self, state: SIAState) -> list[str]:
   191→        actions: list[str] = []
   192→        stats = state.parent_stats
   193→
   194→        brier = stats.get("brier_score", 0.25)
   195→        sharpe = stats.get("sharpe", 0.0)
   196→        trades = stats.get("total_trades", 0)
   197→
   198→        # If too few trades, weights are too restrictive
   199→        if trades < 10:
   200→            actions.append("weight_mutation")
   201→
   202→        # If Brier is bad, harness is the bottleneck
   203→        if brier > 0.25 and self.use_llm:
   204→            actions.append("harness_patch")
   205→
   206→        # If Sharpe is bad, weights need tuning
   207→        if sharpe < 0.5:
   208→            actions.append("weight_mutation")
   209→
   210→        # Always try at least one action
   211→        if not actions:
   212→            actions.append("weight_mutation")
   213→
   214→        return actions

   217→class TargetAgent:
   218→    """Generates the candidate (weight mutation or harness patch).
   219→
   220→    For weight mutations: small +/- 1-3% shifts on random models.
   221→    For harness patches: asks the LLM for a diff to predict_yes_probability.
   222→    """
   223→
   224→    def __init__(self, use_llm: bool, seed: int = 42):
   225→        self.use_llm = use_llm
   226→        self.rng = random.Random(seed)
   227→
   228→    def mutate_weights(self, parent: Hypothesis) -> Hypothesis:
   229→        """Small weight mutation: pick 2 models at random, shift 1-3%."""
   230→        models = list(parent.model_weights.keys())
   231→        if len(models) < 2:
   232→            return parent
   233→
   234→        boost, trim = self.rng.sample(models, 2)
   235→        delta = self.rng.uniform(0.01, 0.03)
   236→
   237→        new_weights = dict(parent.model_weights)
   238→        new_weights[boost] = max(0.01, new_weights[boost] + delta)
   239→        new_weights[trim] = max(0.01, new_weights[trim] - delta)
   240→        new_weights = _normalise(new_weights)
   241→
   242→        # Also randomly nudge min_edge / kelly by a tiny amount
   243→        new_min_edge = max(
   244→            0.01,
   245→            min(0.15, parent.min_edge + self.rng.choice([-0.01, -0.005, 0, 0.005, 0.01])),
   246→        )
   247→        new_kelly = max(
   248→            0.05,
   249→            min(
   250→                0.30,
   251→                parent.kelly_fraction + self.rng.choice([-0.02, -0.01, 0, 0.01, 0.02]),
   252→            ),
   253→        )
   254→
   255→        return Hypothesis(
   256→            description=f"SIA hourly: +{delta:.3f} {boost} / -{delta:.3f} {trim}",
   257→            model_weights=new_weights,
   258→            min_edge=round(new_min_edge, 4),
   259→            kelly_fraction=round(new_kelly, 4),
   260→            max_bet_pct=parent.max_bet_pct,
   261→            tail_filter_enabled=parent.tail_filter_enabled,
   262→            tail_filter_threshold_high=parent.tail_filter_threshold_high,
   263→            tail_filter_threshold_low=parent.tail_filter_threshold_low,
   264→            tail_filter_correction_high=parent.tail_filter_correction_high,
   265→            tail_filter_correction_low=parent.tail_filter_correction_low,
   266→            source="sia_weight_mutation",
   267→        )
   268→
   269→    def propose_harness_patch(self, parent: Hypothesis, stats: dict[str, float]) -> str | None:
   270→        """Ask the LLM for a patched harness source.
   271→
   272→        Returns the new full source code, or None on failure / no LLM.
   273→        Uses the shared ``asi_engine.llm_client`` helper (ZAI / GLM only).
   274→        """
   275→        if not self.use_llm:
   276→            return None
   277→
   278→        current_src = _load_harness_source()
   279→
   280→        prompt = (
   281→            "You are the Target agent in a SIA (Self-Improving Algorithm) loop.\n"
   282→            "Your job: propose a patched version of `predict_yes_probability`.\n\n"
   283→            f"Current source:\n```python\n{current_src}\n```\n\n"
   284→            f"Current parent stats: {json.dumps(stats, indent=2)}\n\n"
   285→            f"Parent weights: {json.dumps(parent.model_weights, indent=2)}\n\n"
   286→            "Requirements:\n"
   287→            "1. Keep the same function signature.\n"
   288→            "2. Return only the FULL patched Python source code, no prose.\n"
   289→            "3. The patched function must still return a float in [0.01, 0.99].\n"
   290→            "4. Try ONE focused improvement (e.g. temperature bias correction, "
   291→            "non-linear ensemble combination, volatility-scaled z-score).\n"
   292→        )
   293→
   294→        raw = chat_json(
   295→            prompt,
   296→            layer="SIA",
   297→            temperature=0.5,
   298→            max_tokens=4000,
   299→        )
   300→        if not raw:
   301→            return None
   302→
   303→        # Strip markdown fences if present
   304→        if "```python" in raw:
   305→            raw = raw.split("```python", 1)[1].split("```", 1)[0]
   306→        elif "```" in raw:
   307→            raw = raw.split("```", 1)[1].split("```", 1)[0]
   308→
   309→        # Sanity check: must contain the function definition
   310→        if "def predict_yes_probability" not in raw:
   311→            logger.warning("LLM patch missing function definition — rejecting")
   312→            return None
   313→
   314→        return raw.strip()

   317→class FeedbackAgent:
   318→    """Evaluates the candidate and decides accept/reject.
   319→
   320→    For weight mutations: uses the standard Layer 1 OOS evaluator.
   321→    For harness patches: writes the patched source to a temp file,
   322→    imports it in a subprocess (sandbox), runs smoke eval, accepts only
   323→    if it beats the incumbent on Sharpe.
   324→    """
   325→
   326→    def __init__(self, brier_df: pd.DataFrame, splits: list[dict[str, Any]]):
   327→        self.brier_df = brier_df
   328→        self.splits = splits
   329→
   330→    def evaluate_weight_mutation(self, hyp: Hypothesis) -> dict[str, float]:
   331→        per_split = [evaluate_hypothesis_oos(self.brier_df, s["test_indices"], hyp) for s in self.splits]
   332→        return _mean_stats(per_split)
   333→
   334→    def evaluate_harness_patch(self, patched_src: str) -> tuple[dict[str, float] | None, str]:
   335→        """Evaluate a harness patch in a subprocess sandbox.
   336→
   337→        LLM-generated code is NEVER executed in the main process.
   338→        Returns (stats, error_message). If stats is None, the patch was
   339→        rejected (syntax error, runtime error, or import failure).
   340→        """
   341→        import subprocess
   342→
   343→        # Write LLM source to a temp file
   344→        src_fd, src_path = tempfile.mkstemp(suffix=".py", prefix="sia_patch_")
   345→        try:
   346→            with os.fdopen(src_fd, "w", encoding="utf-8") as f:
   347→                f.write(patched_src)
   348→
   349→            # Syntax check in main process (cheap, safe — compile() doesn't exec)
   350→            try:
   351→                with open(src_path, encoding="utf-8") as f:
   352→                    compile(f.read(), src_path, "exec")
   353→            except SyntaxError as e:
   354→                return None, f"SyntaxError: {e}"
   355→
   356→            # Prepare input data for subprocess (brier rows for OOS eval)
   357→            brier_rows: list[dict] = []
   358→            if not self.brier_df.empty:
   359→                prob_cols = [c for c in self.brier_df.columns if c.startswith("prob_")]
   360→                if prob_cols:
   361→                    for _, row in self.brier_df.iterrows():
   362→                        realized = row.get("realized_yes")
   363→                        if realized is None or pd.isna(realized):
   364→                            continue
   365→                        r: dict[str, Any] = {"realized_yes": float(realized)}
   366→                        for c in prob_cols:
   367→                            v = row.get(c)
   368→                            if v is not None and not pd.isna(v):
   369→                                r[c] = float(v)
   370→                        brier_rows.append(r)
   371→
   372→            inp_fd, inp_path = tempfile.mkstemp(suffix=".json", prefix="sia_input_")
   373→            try:
   374→                with os.fdopen(inp_fd, "w", encoding="utf-8") as f:
   375→                    json.dump({"brier_rows": brier_rows}, f)
   376→
   377→                # Run smoke test + OOS eval in isolated subprocess
   378→                runner = os.path.join(os.path.dirname(__file__), "_sandbox_runner.py")
   379→                result = subprocess.run(
   380→                    [sys.executable, runner, src_path, inp_path],
   381→                    capture_output=True,
   382→                    text=True,
   383→                    timeout=30,
   384→                )
   385→
   386→                if result.returncode != 0:
   387→                    err = result.stdout.strip() or result.stderr.strip()
   388→                    # Try to parse structured error from JSON output
   389→                    try:
   390→                        err_obj = json.loads(err)
   391→                        return None, err_obj.get("error", err)
   392→                    except (json.JSONDecodeError, AttributeError):
   393→                        return None, f"Subprocess error: {err}"
   394→
   395→                stats = json.loads(result.stdout.strip())
   396→                if not stats:
   397→                    return None, "Empty subprocess output"
   398→                return stats, ""
   399→
   400→            finally:
   401→                try:
   402→                    os.unlink(inp_path)
   403→                except OSError:
   404→                    pass
   405→
   406→        except subprocess.TimeoutExpired:
   407→            return None, "Sandbox timeout (>30s)"
   408→        except Exception as e:
   409→            return None, f"Unexpected error: {e}"
   410→        finally:
   411→            try:
   412→                os.unlink(src_path)
   413→            except OSError:
   414→                pass

   422→def _mean_stats(per_split: list[dict[str, float]]) -> dict[str, float]:
   423→    if not per_split:
   424→        return {
   425→            "sharpe": 0.0,
   426→            "roi_pct": 0.0,
   427→            "win_rate": 0.0,
   428→            "total_trades": 0,
   429→            "brier_score": 0.25,
   430→            "total_pnl": 0.0,
   431→            "total_staked": 0.0,
   432→        }
   433→    keys = ["sharpe", "roi_pct", "win_rate", "brier_score", "total_pnl", "total_staked"]
   434→    out = {k: sum(s.get(k, 0.0) for s in per_split) / len(per_split) for k in keys}
   435→    out["total_trades"] = int(sum(s.get("total_trades", 0) for s in per_split))
   436→    out["sharpe"] = round(out["sharpe"], 4)
   437→    out["roi_pct"] = round(out["roi_pct"], 4)
   438→    out["brier_score"] = round(out["brier_score"], 4)
   439→    out["win_rate"] = round(out["win_rate"], 4)
   440→    return out

   496→
   497→def run_sia_hourly(
   498→    use_llm: bool = False,
   499→    seed: int = 42,
   500→) -> dict[str, Any]:
   501→    """Run one SIA hourly cycle.
   502→
   503→    Returns a summary dict with the cycle's actions and outcomes.
   504→    """
   505→    random.seed(seed)
   506→    _ensure_harness()
   507→
   508→    # 1. Pull unified Brier dataset + splits
   509→    ds = UnifiedDatastore()
   510→    try:
   511→        brier_df = ds.build_brier_dataset()
   512→    except Exception as e:
   513→        logger.warning(
   514→            "build_brier_dataset() raised %s — run polymarket_ingest with the "
   515→            "weather market parser first. Returning early.",
   516→            e,
   517→        )
   518→        return {"error": "brier_dataset_unavailable", "detail": str(e), "cycles_run": 0}
   519→
   520→    if brier_df is None or brier_df.empty:
   521→        logger.error("Brier dataset is empty")
   522→        return {"error": "empty_brier_dataset", "cycles_run": 0}
   523→
   524→    # Add per-model prob columns (tries real forecast join, falls back to synthetic)
   525→    from asi_engine.karpathy_weekly import add_per_model_probabilities
   526→
   527→    brier_df = add_per_model_probabilities(brier_df, ds=ds)
   528→
   529→    splits = ds.build_walk_forward_splits()
   530→    if not splits:
   531→        splits = [
   532→            {
   533→                "split_n": 1,
   534→                "test_indices": brier_df.index.tolist(),
   535→                "train_indices": [],
   536→            }
   537→        ]
   538→
   539→    # 2. Load parent (SIA's own best, or Layer 2 best, or Layer 1 best, or prior)
   540→    parent_hyp, parent_stats = _load_best()
   541→    if parent_hyp is None:
   542→        # Try Layer 2 (ASI-Evolve) best
   543→        from asi_engine.asi_evolve import _load_best as load_asi_evolve_best
   544→        from asi_engine.karpathy_weekly import _load_best as load_karpathy_best
   545→
   546→        conn = None
   547→        try:
   548→            from asi_engine.asi_evolve import _get_db
   549→
   550→            conn = _get_db()
   551→            parent_hyp, parent_stats = load_asi_evolve_best(conn)
   552→        except Exception:
   553→            pass
   554→        finally:
   555→            if conn:
   556→                conn.close()
   557→        if parent_hyp is None:
   558→            parent_hyp = load_karpathy_best()
   559→        if parent_hyp is None:
   560→            parent_hyp = Hypothesis(
   561→                description="Uniform prior (SIA seed)",
   562→                model_weights=_uniform_weights(),
   563→                min_edge=0.05,
   564→                kelly_fraction=0.15,
   565→                max_bet_pct=0.05,
   566→            )
   567→            # Re-eval on current splits
   568→            feedback = FeedbackAgent(brier_df, splits)
   569→            parent_stats = feedback.evaluate_weight_mutation(parent_hyp)
   570→        else:
   571→            # Re-eval Layer 1/2 best on current splits
   572→            feedback = FeedbackAgent(brier_df, splits)
   573→            parent_stats = feedback.evaluate_weight_mutation(parent_hyp)
   574→        _save_best(parent_hyp, parent_stats)
   575→
   576→    state = SIAState(parent_hypothesis=parent_hyp, parent_stats=parent_stats)
   577→
   578→    # 3. Run Meta → Target → Feedback
   579→    meta = MetaAgent(use_llm=use_llm)
   580→    target = TargetAgent(use_llm=use_llm, seed=seed)
   581→    feedback = FeedbackAgent(brier_df, splits)
   582→
   583→    actions = meta.decide(state)
   584→    logger.info("SIA cycle actions: %s", actions)
   585→
   586→    best_hyp = parent_hyp
   587→    best_stats = parent_stats
   588→    actions_taken: list[dict[str, Any]] = []
   589→
   590→    for action in actions:
   591→        if action == "weight_mutation":
   592→            cand_hyp = target.mutate_weights(parent_hyp)
   593→            cand_stats = feedback.evaluate_weight_mutation(cand_hyp)
   594→
   595→            improved = (
   596→                cand_stats["sharpe"] > best_stats.get("sharpe", -1e9)
   597→                and cand_stats["brier_score"] <= best_stats.get("brier_score", 1.0) * 1.05
   598→                and cand_stats["total_trades"] >= 3
   599→            )
   600→
   601→            if improved:
   602→                logger.info(
   603→                    "  [weight_mutation] ✓ sharpe %.3f > %.3f",
   604→                    cand_stats["sharpe"],
   605→                    best_stats.get("sharpe", 0.0),
   606→                )
   607→                best_hyp = cand_hyp
   608→                best_stats = cand_stats
   609→                _save_best(cand_hyp, cand_stats)
   610→                _append_results_tsv(1, "weight_mutation", cand_hyp, cand_stats, "keep")
   611→                actions_taken.append(
   612→                    {
   613→                        "action": "weight_mutation",
   614→                        "status": "keep",
   615→                        "hypothesis": cand_hyp.to_dict(),
   616→                        "stats": cand_stats,
   617→                    }
   618→                )
   619→            else:
   620→                logger.info(
   621→                    "  [weight_mutation] ✗ sharpe %.3f ≤ %.3f",
   622→                    cand_stats["sharpe"],
   623→                    best_stats.get("sharpe", 0.0),
   624→                )
   625→                _append_results_tsv(1, "weight_mutation", cand_hyp, cand_stats, "reject")
   626→                actions_taken.append(
   627→                    {
   628→                        "action": "weight_mutation",
   629→                        "status": "reject",
   630→                        "hypothesis": cand_hyp.to_dict(),
   631→                        "stats": cand_stats,
   632→                    }
   633→                )
   634→
   635→        elif action == "harness_patch":
   636→            patched_src = target.propose_harness_patch(parent_hyp, parent_stats)
   637→            if patched_src is None:
   638→                logger.info("  [harness_patch] skipped (no LLM or proposal failed)")
   639→                actions_taken.append(
   640→                    {
   641→                        "action": "harness_patch",
   642→                        "status": "skipped",
   643→                        "reason": "no_llm_or_proposal_failed",
   644→                    }
   645→                )
   646→                continue
   647→
   648→            cand_stats, err = feedback.evaluate_harness_patch(patched_src)
   649→            if cand_stats is None:
   650→                logger.info("  [harness_patch] ✗ rejected: %s", err)
   651→                _append_results_tsv(
   652→                    1,
   653→                    "harness_patch",
   654→                    parent_hyp,
   655→                    {"sharpe": 0, "roi_pct": 0, "brier_score": 1, "total_trades": 0},
   656→                    "reject",
   657→                    note=err,
   658→                )
   659→                actions_taken.append(
   660→                    {
   661→                        "action": "harness_patch",
   662→                        "status": "reject",
   663→                        "reason": err,
   664→                    }
   665→                )
   666→                continue
   667→
   668→            # Accept harness patch only if Brier improves meaningfully
   669→            improved = cand_stats["brier_score"] < best_stats.get("brier_score", 1.0) * 0.95
   670→            if improved:
   671→                logger.info(
   672→                    "  [harness_patch] ✓ brier %.4f < %.4f — persisting patched harness",
   673→                    cand_stats["brier_score"],
   674→                    best_stats.get("brier_score", 1.0),
   675→                )
   676→                # Persist the patched harness
   677→                with open(HARNESS_PATH, "w", encoding="utf-8") as f:
   678→                    f.write(patched_src)
   679→                # Reload
   680→                import asi_engine.sia_harness as harness_mod
   681→
   682→                importlib.reload(harness_mod)
   683→                _append_results_tsv(1, "harness_patch", parent_hyp, cand_stats, "keep")
   684→                actions_taken.append(
   685→                    {
   686→                        "action": "harness_patch",
   687→                        "status": "keep",
   688→                        "stats": cand_stats,
   689→                        "diff_preview": patched_src[:500],
   690→                    }
   691→                )
   692→            else:
   693→                logger.info(
   694→                    "  [harness_patch] ✗ brier %.4f ≥ %.4f",
   695→                    cand_stats["brier_score"],
   696→                    best_stats.get("brier_score", 1.0) * 0.95,
   697→                )
   698→                _append_results_tsv(1, "harness_patch", parent_hyp, cand_stats, "reject")
   699→                actions_taken.append(
   700→                    {
   701→                        "action": "harness_patch",
   702→                        "status": "reject",
   703→                        "stats": cand_stats,
   704→                    }
   705→                )
   706→
   707→    logger.info(
   708→        "SIA hourly done. Best: sharpe=%.3f brier=%.4f desc=%r",
   709→        best_stats.get("sharpe", 0.0),
   710→        best_stats.get("brier_score", 0.25),
   711→        best_hyp.description,
   712→    )
   713→
   714→    return {
   715→        "cycles_run": 1,
   716→        "actions_taken": actions_taken,
   717→        "best_hypothesis": best_hyp.to_dict(),
   718→        "best_stats": best_stats,
   719→        "n_splits": len(splits),
