     1→"""Layer 1: Karpathy-style weekly autonomous hypothesis engine.
     2→
     3→This module replaces the original `autoresearch/auto_scientist.py` loop which
     4→ran against a *synthetic* `eval_harness.py` (random seed 101, 1000 fake
     5→scenarios). That synthetic harness was the root cause of the inflated
     6→+74.34% ROI claim — it sampled ground-truth temperatures from a Uniform(5, 42)
     7→distribution and built market prices from the forecast ensemble, so the model
     8→could only "win" by exploiting the synthetic noise.
     9→
    10→This Karpathy layer instead uses the **real** unified_datastore:
    11→
    12→  1. Pull `build_brier_dataset()` (markets ⋈ actuals on (city, target_date),
    13→     realized_yes derived from market_type HIGH/LOW vs temperature_2m_max).
    14→  2. Use `build_walk_forward_splits()` to get N train/test windows where
    15→     test[N] is strictly *after* train[N] — no temporal leakage.
    16→  3. Generate a hypothesis (currently: a parameterised mutation of the
    17→     ensemble weight prior + min_edge + kelly_fraction — the same surface
    18→     `auto_scientist.py` mutated, but evaluated honestly).
    19→  4. Evaluate the hypothesis on the test windows only.
    20→  5. If the mean OOS Sharpe beats the incumbent, persist the winner to
    21→     `data/karpathy_best.json` and append a row to `data/karpathy_results.tsv`.
    22→
    23→This is the "weekly" layer in the 3-layer stack — slow, thorough, explores
    24→broadly. ASI-Evolve (daily) and SIA (hourly) layers consume its output.
    25→
    26→The LLM is intentionally optional: if `ZAI_API_KEY` is set
    27→the loop asks the model (ZAI / GLM via the shared ``llm_client`` helper)
    28→for a hypothesis; otherwise it falls back to a deterministic mutation
    29→ladder. This keeps the module runnable in CI without network access.

   122→# ---------------------------------------------------------------------------
   123→# Brier / Sharpe scoring against the unified Brier dataset
   124→# ---------------------------------------------------------------------------
   125→
   126→
   127→def _weighted_mean_prob(row: pd.Series, weights: dict[str, float], models: list[str]) -> float | None:
   128→    """Return ensemble YES probability for one row.
   129→
   130→    Expects `row` to contain columns named f"prob_{model}" for each model in
   131→    `models`. Returns None if any are missing.
   132→    """
   133→    s = 0.0
   134→    wsum = 0.0
   135→    for m in models:
   136→        col = f"prob_{m}"
   137→        if col not in row or pd.isna(row[col]):
   138→            return None
   139→        w = weights.get(m, 0.0)
   140→        s += w * float(row[col])
   141→        wsum += w
   142→    if wsum <= 0:
   143→        return None
   144→    return s / wsum

   147→def evaluate_hypothesis_oos(
   148→    brier_df: pd.DataFrame,
   149→    test_indices: list[int],
   150→    hyp: Hypothesis,
   151→    models: list[str] | None = None,
   152→) -> dict[str, float]:
   153→    """Evaluate one hypothesis on one OOS test window.
   154→
   155→    The Brier dataset is expected to contain:
   156→      - prob_{model} columns (one per model) — per-model YES probability
   157→      - realized_yes — 0/1 ground truth (already derived by unified_datastore)
   158→      - yes_price, no_price — final resolved prices (used as the "market")
   159→      - days_ahead — for the tail-filter
   160→
   161→    Returns dict of {sharpe, roi_pct, win_rate, total_trades, brier_score}.
   162→
   163→    The Sharpe here is per-trade (mean_pnl / std_pnl). To compare to a real
   164→    Polymarket track record, multiply by sqrt(trades_per_day). This is the
   165→    same convention as eval_harness.py's corrected Sharpe (no annualisation
   166→    — annualisation requires a known trade frequency).
   167→    """
   168→    models = models or DEFAULT_MODELS
   169→
   170→    if brier_df.empty or not test_indices:
   171→        return {
   172→            "sharpe": 0.0,
   173→            "roi_pct": 0.0,
   174→            "win_rate": 0.0,
   175→            "total_trades": 0,
   176→            "brier_score": 0.25,
   177→            "total_pnl": 0.0,
   178→            "total_staked": 0.0,
   179→        }
   180→
   181→    test_df = brier_df.loc[brier_df.index.intersection(test_indices)]
   182→    if test_df.empty:
   183→        return {
   184→            "sharpe": 0.0,
   185→            "roi_pct": 0.0,
   186→            "win_rate": 0.0,
   187→            "total_trades": 0,
   188→            "brier_score": 0.25,
   189→            "total_pnl": 0.0,
   190→            "total_staked": 0.0,
   191→        }
   192→
   193→    pnls: list[float] = []
   194→    total_staked = 0.0
   195→    brier_errors: list[float] = []
   196→
   197→    INIT_BANKROLL = 10000.0  # noqa: N806
   198→    bankroll = INIT_BANKROLL
   199→
   200→    for _, row in test_df.iterrows():
   201→        # 1. Build ensemble YES probability from per-model probabilities
   202→        yes_prob = _weighted_mean_prob(row, hyp.model_weights, models)
   203→        if yes_prob is None:
   204→            continue
   205→
   206→        # 2. Optional tail-filter: nudges the forecast mean when extreme temps
   207→        # are detected (mirrors the GFS overshoot correction in auto_scientist).
   208→        if hyp.tail_filter_enabled and "days_ahead" in row:
   209→            # We don't have the raw temperature here — we use the per-model
   210→            # YES probability dispersion as a proxy for "extreme regime".
   211→            # High dispersion ⇒ uncertain regime ⇒ shrink toward 0.5 a bit.
   212→            model_probs = [row.get(f"prob_{m}", 0.5) for m in models]
   213→            try:
   214→                model_probs = [float(p) for p in model_probs if not pd.isna(p)]
   215→            except (TypeError, ValueError):
   216→                model_probs = []
   217→            if model_probs:
   218→                disp = max(model_probs) - min(model_probs)
   219→                if disp > 0.35:
   220→                    yes_prob = 0.5 + (yes_prob - 0.5) * 0.85
   221→
   222→        # 3. Market entry price — we need this to compute Kelly + edge.
   223→        # In production this comes from resolvedmarkets_ingest snapshots.
   224→        # We ONLY use snapshot_yes_price (the pre-resolution orderbook price).
   225→        # The resolved yes_price (0.0/1.0) is NEVER used as entry proxy
   226→        # because it creates look-ahead bias: the model's bet would always
   227→        # appear profitable against the resolved outcome.
   228→        market_yes_price: float | None = None
   229→        if "snapshot_yes_price" in row and not pd.isna(row.get("snapshot_yes_price", float("nan"))):
   230→            market_yes_price = float(row["snapshot_yes_price"])
   231→
   232→        # 4. Brier score (always — even without a market price)
   233→        realized = row.get("realized_yes")
   234→        if realized is None or pd.isna(realized):
   235→            continue
   236→        realized_f = float(realized)
   237→        brier_errors.append((yes_prob - realized_f) ** 2)
   238→
   239→        # 5. Bet decision — only if we have a market price
   240→        if market_yes_price is None:
   241→            continue
   242→        if market_yes_price <= 0.02 or market_yes_price >= 0.98:
   243→            continue  # skip extremes — Kelly degenerates
   244→
   245→        edge = yes_prob - market_yes_price
   246→        if abs(edge) < hyp.min_edge:
   247→            continue
   248→
   249→        # Side selection
   250→        if edge > 0:
   251→            side_yes = True
   252→            entry = market_yes_price
   253→            prob = yes_prob
   254→        else:
   255→            side_yes = False
   256→            entry = 1.0 - market_yes_price
   257→            prob = 1.0 - yes_prob
   258→
   259→        if entry <= 0 or entry >= 1 or prob <= 0 or prob >= 1:
   260→            continue
   261→
   262→        # Kelly sizing (same formula as trading_model.compute_kelly_fraction)
   263→        b = (1.0 / entry) - 1.0
   264→        if b <= 0:
   265→            continue
   266→        q = 1.0 - prob
   267→        f_star = (b * prob - q) / b
   268→        if f_star <= 0:
   269→            continue
   270→
   271→        stake = bankroll * min(f_star * hyp.kelly_fraction, hyp.max_bet_pct)
   272→        if stake <= 0:
   273→            continue
   274→
   275→        total_staked += stake
   276→
   277→        # PnL resolution — with realistic cost model
   278→        #   Polymarket fee: C × feeRate × p × (1-p) at entry time
   279→        #   Slippage: price-impact model — low-liquidity markets (entry < 0.05)
   280→        #     suffer worse fill. Estimated from CLOB orderbook depth:
   281→        #     - High liquidity (entry > 0.10): 0.5% of stake
   282→        #     - Medium (0.05–0.10): 1.0% of stake
   283→        #     - Low liquidity (entry < 0.05): 3.0% of stake (thin books)
   284→        #   Gas / on-chain cost: ~$0.10 flat per trade (Polygon tx)
   285→        GAS_COST_USD = 0.10  # noqa: N806  # Polygon gas per trade
   286→
   287→        # Adaptive slippage based on entry price (liquidity proxy)
   288→        if entry < 0.05:
   289→            SLIPPAGE_PCT = 0.03  # noqa: N806  # thin orderbook
   290→        elif entry < 0.10:
   291→            SLIPPAGE_PCT = 0.01  # noqa: N806  # moderate
   292→        else:
   293→            SLIPPAGE_PCT = 0.005  # noqa: N806  # deep book
   294→
   295→        slippage_cost = stake * SLIPPAGE_PCT
   296→        effective_stake = stake + slippage_cost  # you pay more than planned
   297→        # Effective entry price is worse due to slippage
   298→        effective_entry = entry * (1.0 + SLIPPAGE_PCT)
   299→
   300→        won = (realized_f >= 0.5) == side_yes
   301→        if won:
   302→            gross_payout = effective_stake / effective_entry  # shares at worse price
   303→            # Polymarket taker fee: C × feeRate × p × (1-p)
   304→            shares = effective_stake / effective_entry
   305→            fee = polymarket_fee(shares, effective_entry, WEATHER_FEE_RATE)
   306→            pnl = gross_payout - fee - effective_stake
   307→        else:
   308→            pnl = -effective_stake - GAS_COST_USD  # lose stake + gas
   309→
   310→        pnls.append(pnl)
   311→        # NON-COMPOUNDING: use fixed bankroll to avoid exponential blowup
   312→        # with low-entry-price markets (e.g. 0.02 entry → 50x odds → bankroll explodes).
   313→        # Reset to initial so each trade's Kelly sizing is independent.
   314→        bankroll = INIT_BANKROLL
   315→
   316→    total_trades = len(pnls)
   317→    if total_trades == 0:
   318→        return {
   319→            "sharpe": 0.0,
   320→            "roi_pct": 0.0,
   321→            "win_rate": 0.0,
   322→            "total_trades": 0,
   323→            "brier_score": (sum(brier_errors) / len(brier_errors) if brier_errors else 0.25),
   324→            "total_pnl": 0.0,
   325→            "total_staked": 0.0,
   326→        }
   327→
   328→    wins = sum(1 for p in pnls if p > 0)
   329→    total_pnl = sum(pnls)
   330→    roi_pct = (total_pnl / total_staked) * 100.0 if total_staked > 0 else 0.0
   331→    mean_pnl = total_pnl / total_trades
   332→    var_pnl = sum((p - mean_pnl) ** 2 for p in pnls) / total_trades
   333→    std_pnl = math.sqrt(var_pnl) if var_pnl > 0 else 1e-5
   334→    sharpe = mean_pnl / std_pnl if std_pnl > 0 else 0.0
   335→    brier = sum(brier_errors) / len(brier_errors) if brier_errors else 0.25
   336→
   337→    return {
   338→        "sharpe": round(sharpe, 4),
   339→        "roi_pct": round(roi_pct, 4),
   340→        "win_rate": round(wins / total_trades, 4),
   341→        "total_trades": total_trades,
   342→        "brier_score": round(brier, 4),
   343→        "total_pnl": round(total_pnl, 2),
   344→        "total_staked": round(total_staked, 2),
   345→    }

   353→def _uniform_weights() -> dict[str, float]:
   354→    n = len(DEFAULT_MODELS)
   355→    return {m: round(1.0 / n, 4) for m in DEFAULT_MODELS}

   358→def _normalise(weights: dict[str, float]) -> dict[str, float]:
   359→    s = sum(weights.values())
   360→    if s <= 0:
   361→        return _uniform_weights()
   362→    return {m: round(w / s, 4) for m, w in weights.items()}

   512→def generate_hypothesis(round_num: int, parent: Hypothesis | None = None) -> Hypothesis:
   513→    """Generate one hypothesis from the mutation ladder.
   514→
   515→    If a parent is supplied, the mutation is applied on top of the parent's
   516→    weights/params; otherwise the parent is the uniform prior.
   517→    """
   518→    parent = parent or Hypothesis(
   519→        description="Uniform prior",
   520→        model_weights=_uniform_weights(),
   521→        min_edge=0.05,
   522→        kelly_fraction=0.15,
   523→        max_bet_pct=0.05,
   524→    )
   525→
   526→    mutation = _MUTATION_LADDER[round_num % len(_MUTATION_LADDER)]
   527→
   528→    new_weights = dict(parent.model_weights)
   529→    for m, delta in mutation["weights_delta"].items():
   530→        new_weights[m] = max(0.01, new_weights.get(m, 0.125) + delta)
   531→    new_weights = _normalise(new_weights)
   532→
   533→    new_min_edge = max(0.01, min(0.15, parent.min_edge + mutation["min_edge_delta"]))
   534→    new_kelly = max(0.05, min(0.30, parent.kelly_fraction + mutation["kelly_delta"]))
   535→    new_max_bet = max(0.01, min(0.10, parent.max_bet_pct + mutation.get("max_bet_pct_delta", 0.0)))
   536→
   537→    return Hypothesis(
   538→        description=mutation["description"],
   539→        model_weights=new_weights,
   540→        min_edge=round(new_min_edge, 4),
   541→        kelly_fraction=round(new_kelly, 4),
   542→        max_bet_pct=round(new_max_bet, 4),
   543→        tail_filter_enabled=mutation.get("tail_filter_enabled", parent.tail_filter_enabled),
   544→        tail_filter_threshold_high=parent.tail_filter_threshold_high,
   545→        tail_filter_threshold_low=parent.tail_filter_threshold_low,
   546→        tail_filter_correction_high=parent.tail_filter_correction_high,
   547→        tail_filter_correction_low=parent.tail_filter_correction_low,
   548→        source="mutation_ladder",
   549→    )

   709→def _load_cached_forecasts(
   710→    ds: UnifiedDatastore,
   711→    brier_df: pd.DataFrame,
   712→) -> tuple[pd.DataFrame, set[tuple[str, str]]]:
   713→    """Load cached forecasts from the unified datastore.
   714→
   715→    Reads the forecasts table, filters to temperature_2m_max variable,
   716→    and returns the filtered DataFrame along with the set of (city, date)
   717→    pairs already in cache.
   718→
   719→    Returns:
   720→        (forecasts_df, cached_pairs) where cached_pairs is a set of
   721→        (city_str, join_date_str) tuples.
   722→    """
   723→    forecasts_df = ds.read_forecasts()
   724→    cached_pairs: set[tuple[str, str]] = set()
   725→    if forecasts_df.empty or "target_date" not in forecasts_df.columns:
   726→        return forecasts_df, cached_pairs
   727→
   728→    forecasts_df = forecasts_df.copy()
   729→    forecasts_df["join_date"] = pd.to_datetime(forecasts_df["target_date"], utc=True, errors="coerce").dt.date.astype(
   730→        str
   731→    )
   732→    # FIX: Use explicit column check instead of .get() with default — .get() returns
   733→    # Series | scalar, which confuses pyright about downstream DataFrame typing.
   734→    if "variable" in forecasts_df.columns:
   735→        forecasts_df["variable_key"] = forecasts_df["variable"].fillna("temperature_2m_max")
   736→    else:
   737→        forecasts_df["variable_key"] = "temperature_2m_max"
   738→    mask = forecasts_df["variable_key"].astype(str).str.contains("max", na=False)
   739→    forecasts_df = forecasts_df.loc[mask].copy()
   740→    if not forecasts_df.empty:
   741→        cached_pairs = set(
   742→            zip(
   743→                forecasts_df["city"].astype(str),
   744→                forecasts_df["join_date"].astype(str),
   745→            )
   746→        )
   747→    return forecasts_df, cached_pairs

   750→def _backfill_missing_forecasts(
   751→    ds: UnifiedDatastore,
   752→    brier_df: pd.DataFrame,
   753→    forecasts_df: pd.DataFrame,
   754→    cached_pairs: set[tuple[str, str]],
   755→    needed_pairs: set[tuple[str, str]],
   756→) -> pd.DataFrame:
   757→    """Fetch missing forecasts from Open-Meteo Historical Forecast API.
   758→
   759→    Compares needed (city, date) pairs against the cached set and fetches
   760→    any missing ones from the free Historical Forecast API. Appends the
   761→    fetched data to forecasts_df and persists to disk for future runs.
   762→
   763→    Returns:
   764→        Updated forecasts_df with fetched rows appended.
   765→    """
   766→    missing_pairs = needed_pairs - cached_pairs
   767→    if not missing_pairs:
   768→        return forecasts_df
   769→
   770→    logger.info(
   771→        "Forecast join: %d (city, date) pairs needed, %d cached, %d missing — fetching from Historical Forecast API",
   772→        len(needed_pairs),
   773→        len(cached_pairs),
   774→        len(missing_pairs),
   775→    )
   776→    try:
   777→        from data_pipeline.weather_ensemble import (
   778→            fetch_historical_forecast_ensemble,
   779→        )
   780→
   781→        missing_by_city: dict[str, list[str]] = {}
   782→        for c, d in missing_pairs:
   783→            missing_by_city.setdefault(c, []).append(d)
   784→
   785→        city_coords: dict[str, tuple[float, float]] = {}
   786→        for _, row in brier_df.iterrows():
   787→            c = str(row.get("city", ""))
   788→            if c and c not in city_coords:
   789→                lat = row.get("latitude")
   790→                lon = row.get("longitude")
   791→                try:
   792→                    if lat is not None and lon is not None:
   793→                        city_coords[c] = (float(lat), float(lon))
   794→                except (TypeError, ValueError):
   795→                    pass
   796→
   797→        fetched_frames = []
   798→        for city, dates in missing_by_city.items():
   799→            coords = city_coords.get(city)
   800→            if not coords:
   801→                continue
   802→            lat, lon = coords
   803→            start_date = min(dates)
   804→            end_date = max(dates)
   805→            try:
   806→                df = fetch_historical_forecast_ensemble(
   807→                    lat,
   808→                    lon,
   809→                    start_date=start_date,
   810→                    end_date=end_date,
   811→                    city=city,
   812→                )
   813→                if not df.empty:
   814→                    fetched_frames.append(df)
   815→            except Exception as exc:
   816→                logger.warning(
   817→                    "Historical forecast fetch failed for %s %s..%s: %s",
   818→                    city,
   819→                    start_date,
   820→                    end_date,
   821→                    exc,
   822→                )
   823→
   824→        if fetched_frames:
   825→            fetched_df = pd.concat(fetched_frames, ignore_index=True)
   826→            fetched_df["join_date"] = pd.to_datetime(fetched_df["date"], utc=True, errors="coerce").dt.date.astype(str)
   827→            fetched_df["target_date"] = pd.to_datetime(fetched_df["date"], utc=True, errors="coerce")
   828→            fetched_df["fetched_at"] = pd.Timestamp.utcnow()
   829→            # FIX: Use explicit column check instead of .get() with default — same
   830→            # pattern as _load_cached_forecasts; avoids pyright Series|scalar ambiguity.
   831→            if "variable" in fetched_df.columns:
   832→                fetched_df["variable_key"] = fetched_df["variable"].fillna("temperature_2m_max")
   833→            else:
   834→                fetched_df["variable_key"] = "temperature_2m_max"
   835→            mask = fetched_df["variable_key"].astype(str).str.contains("max", na=False)
   836→            fetched_df = fetched_df.loc[mask].copy()
   837→            cols = [
   838→                "city",
   839→                "latitude",
   840→                "longitude",
   841→                "target_date",
   842→                "model",
   843→                "variable",
   844→                "value",
   845→                "fetched_at",
   846→                "join_date",
   847→            ]
   848→            # FIX: Explicit DataFrame construction — list-indexing on a DataFrame
   849→            # always returns a DataFrame, but pyright can't infer that. Use .loc[:, :]
   850→            available_cols = [c for c in cols if c in fetched_df.columns]
   851→            fetched_df = fetched_df.loc[:, available_cols].copy()
   852→            if forecasts_df.empty:
   853→                forecasts_df = fetched_df
   854→            else:
   855→                forecasts_df = pd.concat([forecasts_df, fetched_df], ignore_index=True)
   856→            try:
   857→                ds.write_forecasts(forecasts_df.drop(columns=["join_date"], errors="ignore"))
   858→            except Exception as exc:
   859→                logger.warning("Failed to persist fetched forecasts: %s", exc)
   860→            logger.info(
   861→                "Fetched %d rows from Historical Forecast API across %d cities",
   862→                len(fetched_df),
   863→                len(missing_by_city),
   864→            )
   865→    except ImportError:
   866→        logger.warning("data_pipeline.weather_ensemble not importable — cannot dynamically fetch missing forecasts")
   867→    return forecasts_df

   870→def _pivot_and_convert_to_probs(
   871→    brier_df: pd.DataFrame,
   872→    forecasts_df: pd.DataFrame,
   873→    cached_pairs: set[tuple[str, str]],
   874→    needed_pairs: set[tuple[str, str]],
   875→) -> tuple[pd.DataFrame, bool, str]:
   876→    """Pivot forecasts wide and convert temperatures → YES probabilities.
   877→
   878→    Joins the per-model forecasts with the brier dataset and applies a
   879→    normal CDF (sigma=2.0) to convert each model's temperature forecast
   880→    into a YES probability against the market threshold.
   881→
   882→    Returns:
   883→        (brier_df, forecasts_joined, join_origin)
   884→    """
   885→    forecasts_joined = False
   886→    join_origin = "none"
   887→
   888→    if forecasts_df.empty:
   889→        return brier_df, forecasts_joined, join_origin
   890→
   891→    forecasts_df = forecasts_df.copy()
   892→    forecasts_df["model_internal"] = forecasts_df["model"].map(lambda m: OPEN_METEO_API_TO_INTERNAL.get(m, m))
   893→    temp_pivot = forecasts_df.pivot_table(
   894→        index=["city", "join_date"],
   895→        columns="model_internal",
   896→        values="value",
   897→        aggfunc="mean",
   898→    ).reset_index()
   899→    merged = brier_df.merge(temp_pivot, on=["city", "join_date"], how="left", suffixes=("", "_fc"))
   900→
   901→    sigma = 2.0
   902→    for model in DEFAULT_MODELS:
   903→        if model not in merged.columns:
   904→            continue
   905→        col = f"prob_{model}"
   906→
   907→        def _to_prob(forecast, mt, thresh, sigma=sigma):
   908→            if pd.isna(forecast) or pd.isna(thresh) or thresh is None:
   909→                return float("nan")
   910→            try:
   911→                forecast = float(forecast)
   912→                thresh = float(thresh)
   913→            except (TypeError, ValueError):
   914→                return float("nan")
   915→            z = (forecast - thresh) / sigma
   916→            p_high = 0.5 * (1.0 + math.erf(z / math.sqrt(2.0)))
   917→            if str(mt).upper() == "LOW":
   918→                return 1.0 - p_high
   919→            return p_high
   920→
   921→        merged[col] = [
   922→            _to_prob(row.get(model), row.get("market_type"), row.get("threshold")) for _, row in merged.iterrows()
   923→        ]
   924→
   925→    brier_df = merged
   926→    prob_cols_after = [c for c in brier_df.columns if c.startswith("prob_")]
   927→    if prob_cols_after:
   928→        non_null_counts = brier_df[prob_cols_after].notna().sum().sum()
   929→        forecasts_joined = non_null_counts > 0
   930→        if forecasts_joined:
   931→            join_origin = "cached_table" if cached_pairs and cached_pairs >= needed_pairs else "historical_api"
   932→        logger.info(
   933→            "Forecast join %s (origin=%s, %d non-null prob values across %d rows)",
   934→            "succeeded" if forecasts_joined else "failed (no overlapping rows)",
   935→            join_origin,
   936→            non_null_counts,
   937→            len(brier_df),
   938→        )
   939→
   940→    return brier_df, forecasts_joined, join_origin

   943→def add_per_model_probabilities(
   944→    brier_df: pd.DataFrame,
   945→    ds: UnifiedDatastore | None = None,
   946→    *,
   947→    fetch_missing: bool = True,
   948→) -> pd.DataFrame:
   949→    """Add prob_{model} columns to a Brier dataset using real forecast data only.
   950→
   951→    Uses the real forecast join path:
   952→      1. Read unified_forecasts table (populated by
   953→         data_pipeline.unified_datastore.ingest_all() step [3/4]).
   954→      2. If that table is empty OR no overlap with the Brier dataset's
   955→         (city, target_date) pairs AND fetch_missing=True, dynamically
   956→         fetch per-model historical forecasts from Open-Meteo Historical
   957→         Forecast API (free, no key) for every missing (city, date).
   958→      3. Convert per-model temperatures → YES probabilities via a normal
   959→         CDF with sigma=2.0 against the market threshold.
   960→
   961→    If the real forecast join fails (no overlapping rows), prob_ columns
   962→    are NOT added. Downstream code will naturally skip rows without
   963→    real forecast data.
   964→
   965→    A `forecast_join_origin` column is added so downstream code can tell
   966→    whether the per-model probabilities came from real forecasts
   967→    ("historical_api" / "cached_table") or not ("none").
   968→
   969→    This is extracted from run_karpathy_weekly() so Layer 2 (ASI-Evolve)
   970→    and Layer 3 (SIA) can reuse it without re-running the join logic.
   971→    """
   972→    if any(c.startswith("prob_") for c in brier_df.columns):
   973→        return brier_df
   974→
   975→    if ds is None:
   976→        ds = UnifiedDatastore()
   977→
   978→    brier_df = brier_df.copy()
   979→    brier_df["join_date"] = pd.to_datetime(brier_df["target_date"], utc=True, errors="coerce").dt.date.astype(str)
   980→
   981→    # 1. Load cached forecasts
   982→    forecasts_df, cached_pairs = _load_cached_forecasts(ds, brier_df)
   983→
   984→    # 2. Compute needed pairs (used by backfill + pivot)
   985→    needed_pairs = set(
   986→        zip(
   987→            brier_df["city"].astype(str),
   988→            brier_df["join_date"].astype(str),
   989→        )
   990→    )
   991→
   992→    # 3. Backfill missing forecasts from API
   993→    if fetch_missing and not brier_df.empty:
   994→        forecasts_df = _backfill_missing_forecasts(ds, brier_df, forecasts_df, cached_pairs, needed_pairs)
   995→
   996→    # 4. Pivot and convert to probabilities (real data only)
   997→    brier_df, forecasts_joined, join_origin = _pivot_and_convert_to_probs(
   998→        brier_df, forecasts_df, cached_pairs, needed_pairs
   999→    )
  1000→
  1001→    # 5. If real join failed — log and return without prob_ columns
  1002→    if not forecasts_joined:
  1003→        logger.warning(
  1004→            "Real forecast join produced no overlapping (city, date) rows. "
  1005→            "No prob_ columns added. Downstream code will skip these rows."
  1006→        )
  1007→
  1008→    brier_df["forecast_join_origin"] = join_origin
  1009→    return brier_df

  1013→    rounds: int = 6,
  1014→    use_llm: bool = False,
  1015→    seed: int = 42,
  1016→) -> dict[str, Any]:
  1017→    """Run the weekly Karpathy loop.
  1018→
  1019→    For each round:
  1020→      1. Pull the latest unified Brier dataset.
  1021→      2. Build walk-forward splits.
  1022→      3. Propose a hypothesis (LLM if enabled, else mutation ladder).
  1023→      4. Evaluate it on each split's test window.
  1024→      5. Mean the per-split stats → mean_sharpe / mean_roi / mean_brier.
  1025→      6. If mean_sharpe > incumbent, persist as new best.
  1026→
  1027→    Args:
  1028→        rounds: number of hypotheses to test this run.
  1029→        use_llm: if True, ask the LLM for proposals (falls back to ladder on error).
  1030→        seed: RNG seed for reproducibility.
  1031→
  1032→    Returns a summary dict with the best hypothesis and its stats.
  1033→    """
  1034→    random.seed(seed)
  1035→    ds = UnifiedDatastore()
  1036→
  1037→    # 1. Pull Brier dataset (markets ⋈ actuals)
  1038→    try:
  1039→        brier_df = ds.build_brier_dataset()
  1040→    except Exception as e:
  1041→        logger.warning(
  1042→            "build_brier_dataset() raised %s — the unified markets table does "
  1043→            "not yet have parsed weather fields (city/target_date/market_type). "
  1044→            "Run polymarket_ingest with the weather market parser first. "
  1045→            "Returning early without running any rounds.",
  1046→            e,
  1047→        )
  1048→        return {
  1049→            "error": "brier_dataset_unavailable",
  1050→            "detail": str(e),
  1051→            "rounds_run": 0,
  1052→        }
  1053→    if brier_df is None or brier_df.empty:
  1054→        logger.error(
  1055→            "Brier dataset is empty — run data_pipeline.ingest_all() with the "
  1056→            "weather market parser to populate markets.city / target_date / "
  1057→            "market_type / threshold before running this layer."
  1058→        )
  1059→        return {"error": "empty_brier_dataset", "rounds_run": 0}
  1060→
  1061→    # Ensure per-model prob columns exist (tries real forecast join first,
  1062→    # falls back to synthetic per-model probabilities).
  1063→    brier_df = add_per_model_probabilities(brier_df, ds=ds)
  1064→
  1065→    # 2. Walk-forward splits
  1066→    splits = ds.build_walk_forward_splits()
  1067→    if not splits:
  1068→        logger.warning(
  1069→            "No walk-forward splits built — falling back to a single all-data "
  1070→            "evaluation. This is NOT OOS but allows the loop to still run."
  1071→        )
  1072→        splits = [
  1073→            {
  1074→                "split_n": 1,
  1075→                "test_indices": brier_df.index.tolist(),
  1076→                "train_indices": [],
  1077→                "test_start": (brier_df["target_date"].min() if "target_date" in brier_df else None),
  1078→                "test_end": (brier_df["target_date"].max() if "target_date" in brier_df else None),
  1079→            }
  1080→        ]
  1081→
  1082→    # 3. Load incumbent
  1083→    incumbent = _load_best()
  1084→    incumbent_stats: dict[str, float] | None = None
  1085→    if incumbent:
  1086→        # Re-evaluate incumbent on the current splits for a fair comparison
  1087→        per_split = [evaluate_hypothesis_oos(brier_df, s["test_indices"], incumbent) for s in splits]
  1088→        incumbent_stats = _mean_stats(per_split)
  1089→        logger.info(
  1090→            "Loaded incumbent: sharpe=%.3f roi=%.2f%% brier=%.4f",
  1091→            incumbent_stats.get("sharpe", 0.0),
  1092→            incumbent_stats.get("roi_pct", 0.0),
  1093→            incumbent_stats.get("brier_score", 0.25),
  1094→        )
  1095→    else:
  1096→        incumbent = Hypothesis(
  1097→            description="Uniform prior (baseline)",
  1098→            model_weights=_uniform_weights(),
  1099→            min_edge=0.05,
  1100→            kelly_fraction=0.15,
  1101→            max_bet_pct=0.05,
  1102→        )
  1103→        logger.info("No incumbent — starting from uniform prior")
  1104→
  1105→    best_hyp = incumbent
  1106→    best_stats = incumbent_stats or {
  1107→        "sharpe": -1e9,
  1108→        "roi_pct": -1e9,
  1109→        "brier_score": 1.0,
  1110→        "win_rate": 0.0,
  1111→        "total_trades": 0,
  1112→        "total_pnl": 0.0,
  1113→        "total_staked": 0.0,
  1114→    }
  1115→
  1116→    # 4. Run rounds
  1117→    for r in range(1, rounds + 1):
  1118→        logger.info("--- Karpathy round %d/%d ---", r, rounds)
  1119→
  1120→        # Propose hypothesis
  1121→        hyp: Hypothesis | None = None
  1122→        if use_llm:
  1123→            hyp = llm_propose_hypothesis(
  1124→                best_hyp,
  1125→                context={"last_stats": best_stats, "insights": []},
  1126→            )
  1127→        if hyp is None:
  1128→            hyp = generate_hypothesis(r - 1, parent=best_hyp)
  1129→
  1130→        logger.info("Hypothesis: %s (source=%s)", hyp.description, hyp.source)
  1131→
  1132→        # Evaluate on each split's test window
  1133→        per_split = [evaluate_hypothesis_oos(brier_df, s["test_indices"], hyp) for s in splits]
  1134→        mean_stats = _mean_stats(per_split)
  1135→        logger.info(
  1136→            "  OOS mean: sharpe=%.3f roi=%.2f%% brier=%.4f trades=%d",
  1137→            mean_stats["sharpe"],
  1138→            mean_stats["roi_pct"],
  1139→            mean_stats["brier_score"],
  1140→            mean_stats["total_trades"],
  1141→        )
  1142→
  1143→        # Acceptance: must beat incumbent on Sharpe (and not blow up Brier)
  1144→        improved = mean_stats["sharpe"] > best_stats["sharpe"] and (
  1145→            mean_stats["brier_score"] <= best_stats["brier_score"] * 1.10
  1146→        )
  1147→        if improved and mean_stats["total_trades"] >= 5:
  1148→            logger.info("  ✓ ACCEPTED — new best")
  1149→            best_hyp = hyp
  1150→            best_stats = mean_stats
  1151→            _save_best(hyp, mean_stats)
  1152→            _append_results_tsv(r, hyp, mean_stats, "keep")
  1153→        else:
  1154→            logger.info(
  1155→                "  ✗ rejected (sharpe %.3f ≤ incumbent %.3f)",
  1156→                mean_stats["sharpe"],
  1157→                best_stats["sharpe"],
  1158→            )
  1159→            _append_results_tsv(r, hyp, mean_stats, "reject")
  1160→
  1161→        # Small delay to be friendly to the LLM rate limit
  1162→        if use_llm:
  1163→            time.sleep(1.0)
  1164→
  1165→    logger.info(
  1166→        "Karpathy weekly done. Best: sharpe=%.3f roi=%.2f%% desc=%r",
  1167→        best_stats.get("sharpe", 0.0),
  1168→        best_stats.get("roi_pct", 0.0),
  1169→        best_hyp.description,
  1170→    )
  1171→
  1172→    return {
  1173→        "rounds_run": rounds,
  1174→        "best_hypothesis": best_hyp.to_dict(),
  1175→        "best_stats": best_stats,
  1176→        "n_splits": len(splits),
  1177→        "brier_dataset_rows": len(brier_df),
  1178→    }
  1179→
  1180→
  1181→def _mean_stats(per_split: list[dict[str, float]]) -> dict[str, float]:
  1182→    """Mean across splits, weighting each split equally."""
  1183→    if not per_split:
  1184→        return {
  1185→            "sharpe": 0.0,
  1186→            "roi_pct": 0.0,
  1187→            "win_rate": 0.0,
  1188→            "total_trades": 0,
  1189→            "brier_score": 0.25,
  1190→            "total_pnl": 0.0,
  1191→            "total_staked": 0.0,
  1192→        }
  1193→    keys = ["sharpe", "roi_pct", "win_rate", "brier_score", "total_pnl", "total_staked"]
  1194→    out = {k: sum(s.get(k, 0.0) for s in per_split) / len(per_split) for k in keys}
  1195→    out["total_trades"] = int(sum(s.get("total_trades", 0) for s in per_split))
  1196→    out["sharpe"] = round(out["sharpe"], 4)
  1197→    out["roi_pct"] = round(out["roi_pct"], 4)
  1198→    out["brier_score"] = round(out["brier_score"], 4)
  1199→    out["win_rate"] = round(out["win_rate"], 4)
  1200→    return out
